543 bool Dssd4SHEProcessor::pickEventType(SheEvent& event) {
562  int condition = 0;
563  if (event.get_beam())
564  condition += 1;
565  if (event.get_mwpc() > 0)
566  condition += 2;
567  if (event.get_veto())
568  condition += 4;
569 
570  if (condition == 0) {
571  double energy = event.get_energy();
572  if (energy < highEnergyCut_)
573  event.set_type(alpha);
574  else if (energy < fissionEnergyCut_)
575  event.set_type(unknown);
576  else
577  event.set_type(fission);
578  }
579  else if (condition == 1) {
580  double energy = event.get_energy();
581  if (energy < lowEnergyCut_)
582  event.set_type(unknown);
583  else if (energy < highEnergyCut_)
584  event.set_type(alpha);
585  else if (energy < fissionEnergyCut_)
586  event.set_type(unknown);
587  else
588  event.set_type(fission);
589  }
590  else if (condition == 2 ||
591  condition == 4 ||
592  condition == 6) {
593  event.set_type(unknown);
594  }
595  else if (condition == 3) {
596  event.set_type(heavyIon);
597  }
598  else if (condition == 5 || condition == 7) {
599  event.set_type(lightIon);
600  }
601  else
602  event.set_type(unknown);
603 
604  return true;
605 }
