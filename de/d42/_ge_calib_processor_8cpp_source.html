181 bool GeCalibProcessor::PreProcess(RawEvent &event) {
182  if (!EventProcessor::PreProcess(event))
183  return false;
184 
185  using namespace dammIds::ge;
186 
187  // Clear all events stored in vectors from previous event
188  geEvents_.clear();
189 
190  static const vector<ChanEvent*> &highEvents = event.GetSummary("ge:clover_high", true)->GetList();
191  static const vector<ChanEvent*> &lowEvents = event.GetSummary("ge:clover_low", true)->GetList();
192 
196  for (vector<ChanEvent*>::const_iterator itLow = lowEvents.begin();
197  itLow != lowEvents.end(); ++itLow) {
198  plot(calib::D_ENERGY_LOWGAIN, (*itLow)->GetCalEnergy());
199  }
200 
201  for (vector<ChanEvent*>::const_iterator itHigh = highEvents.begin();
202  itHigh != highEvents.end(); itHigh++) {
203  int location = (*itHigh)->GetChanID().GetLocation();
204  plot(calib::D_ENERGY_HIGHGAIN, (*itHigh)->GetCalEnergy());
205 
206  if ( (*itHigh)->IsSaturated() || (*itHigh)->IsPileup() )
207  continue;
208 
209  // find the matching low gain event
210  vector <ChanEvent*>::const_iterator itLow = lowEvents.begin();
211  for (; itLow != lowEvents.end(); itLow++) {
212  if ( (*itLow)->GetChanID().GetLocation() == location ) {
213  break;
214  }
215  }
216  if ( itLow != lowEvents.end() ) {
217  double ratio = (*itHigh)->GetEnergy() / (*itLow)->GetEnergy();
218  plot(calib::DD_CLOVER_ENERGY_RATIO, location, ratio * 10.);
219  if ( (ratio < lowRatio_ ||
220  ratio > highRatio_) )
221  continue;
222  }
223  geEvents_.push_back(*itHigh);
224  }
225 
226  // now we sort the germanium events according to their corrected time
227  sort(geEvents_.begin(), geEvents_.end(), CompareCorrectedTime);
228 
229  return true;
230 }
