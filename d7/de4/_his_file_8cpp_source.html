718 void OutputHisFile::Flush(){
719  if(debug_mode)
720  std::cout << "debug: Flushing histogram entries to file.\n";
721 
722  if(writable){ // Do the filling
723  for(std::vector<fill_queue*>::iterator iter = fills_waiting.begin();
724  iter != fills_waiting.end(); iter++){
725 
726  if(!(*iter)->good)
727  continue;
728 
729  current_entry = (*iter)->entry;
730  current_entry->good_counts++;
731 
732  // Seek to the specified bin
733  ofile.seekg(current_entry->offset*2 + (*iter)->byte, std::ios::beg); // input offset
734 
735  unsigned short sval = 0;
736  unsigned int ival = 0;
737 
738  // Overwrite the bin value
739  if(current_entry->use_int){
740  // Get the original value of the bin
741  ofile.read((char*)&ival, 4);
742  ival += (*iter)->weight;
743 
744  // Set the new value of the bin
745  ofile.seekp(current_entry->offset*2 + (*iter)->byte, std::ios::beg); // output offset
746  ofile.write((char*)&ival, 4);
747  }
748  else{
749  // Get the original value of the bin
750  ofile.read((char*)&sval, 2);
751  sval += (short)(*iter)->weight;
752 
753  // Set the new value of the bin
754  ofile.seekp(current_entry->offset*2 + (*iter)->byte, std::ios::beg); // output offset
755  ofile.write((char*)&sval, 2);
756  }
757  }
758  }
759  else if(debug_mode){ std::cout << "debug: Output file is not writable!\n"; }
760 
761  // Delete the drr_entries in the fill_queue vector
762  for(std::vector<fill_queue*>::iterator iter = fills_waiting.begin(); iter != fills_waiting.end(); iter++){
763  delete (*iter);
764  }
765  fills_waiting.clear();
766 
767  Flush_count = 0;
768 }
