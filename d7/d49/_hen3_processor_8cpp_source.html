47 EventData Hen3Processor::BestBetaForNeutron(double nTime) {
48  PlaceOR* betas = dynamic_cast<PlaceOR*>(
49  TreeCorrelator::get()->place("Beta"));
50  unsigned sz = betas->info_.size();
51 
52  if (sz == 0)
53  return EventData(-1);
54 
55  double bestTime = numeric_limits<double>::max();
56  unsigned bestIndex = -1;
57  for (unsigned index = 0; index < sz; ++index) {
58  double dtime = (nTime - betas->info_.at(index).time);
59  // We assume that neutron must be slower than beta
60  if (dtime < 0)
61  continue;
62  if (dtime < bestTime) {
63  bestTime = dtime;
64  bestIndex = index;
65  }
66  }
67  if (bestIndex > 0)
68  return betas->info_.at(bestIndex);
69  else
70  return EventData(-1);
71 }
