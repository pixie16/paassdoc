203  /* Number of neutrons as selected by gates on 3hen spectrum.
204  * See DetectorDriver::InitCorrelator for gates. */

206  dynamic_cast<PlaceCounter*>(
207  TreeCorrelator::get()->place("Neutrons"))->getCounter();
208 
212  if (neutron_count < 1)
213  return true;

220  for (vector<ChanEvent*>::iterator it = geEvents_.begin();
221  it != geEvents_.end(); ++it) {
222  ChanEvent* chan = *it;
223  double gEnergy = chan->GetCalEnergy();
224  if (gEnergy < gammaThreshold_)

227  }
228  return true;
229  }

231  double clockInSeconds = Globals::get()->clockInSeconds();
232 
234  double cycleTime = TreeCorrelator::get()->place("Cycle")->last().time;
235 
236  // beamOn is true for beam on and false for beam off
237  bool beamOn = TreeCorrelator::get()->place("Beam")->status();
238 
239  plot(neutron::D_MULT, geEvents_.size());
240 
241  /* Beta places are activated with threshold in ScintProcessor. */
242  bool hasBeta = TreeCorrelator::get()->place("Beta")->status();
243 
244  for (vector<ChanEvent*>::iterator it1 = geEvents_.begin();
245  it1 != geEvents_.end(); ++it1) {
246  ChanEvent *chan = *it1;
247 
248  double gEnergy = chan->GetCalEnergy();
249  double gTime = chan->GetCorrectedTime();
250  double decayTime = (gTime - cycleTime) * clockInSeconds;
251  if (gEnergy < gammaThreshold_)
252  continue;
253 
254  plot(neutron::D_ENERGY, gEnergy);
255  granploty(neutron::DD_ENERGY__TIMEX,
256  gEnergy, decayTime, timeResolution);
257 
258  if (beamOn) {
259  granploty(neutron::DD_ENERGY__TIMEX_GROW,
260  gEnergy, decayTime, timeResolution);
261  } else {
262  double decayTimeOff = (gTime -
263  TreeCorrelator::get()->place("Beam")->last().time) *
264  clockInSeconds;
265  granploty(neutron::DD_ENERGY__TIMEX_DECAY,
266  gEnergy, decayTimeOff, timeResolution);
267  }
268 
269  double gb_dtime = numeric_limits<double>::max();
270  if (hasBeta) {
271  EventData bestBeta = BestBetaForGamma(gTime);
272  gb_dtime = (gTime - bestBeta.time) * clockInSeconds;
273  plot(neutron::betaGated::D_ENERGY, gEnergy);
274  if (GoodGammaBeta(gb_dtime)) {
275  plot(neutron::betaGated::D_ENERGY_PROMPT, gEnergy);
276  granploty(neutron::betaGated::DD_ENERGY__TIMEX,
277  gEnergy, decayTime, timeResolution);
278  }
279  }
280 
281  for (unsigned l = 0; l < 48; ++l) {
282  stringstream neutron;
283  neutron << "Neutron_" << l;
284  if (TreeCorrelator::get()->place(neutron.str())->status()) {
285  plot(neutron::DD_ENERGY_NEUTRON_LOC, gEnergy, l);
286  }
287  }
288 
289  if (neutron_count > 1) {
290  plot(multiNeutron::D_ENERGY, gEnergy);
291  granploty(multiNeutron::DD_ENERGY__TIMEX,
292  gEnergy, decayTime, timeResolution);
293  if (hasBeta) {
294  plot(multiNeutron::betaGated::D_ENERGY, gEnergy);
295  if (GoodGammaBeta(gb_dtime)) {
296  plot(multiNeutron::betaGated::D_ENERGY_PROMPT, gEnergy);
297  granploty(multiNeutron::betaGated::DD_ENERGY__TIMEX,
298  gEnergy, decayTime, timeResolution);
299  }
300  }
301  }
302 
303  for (vector<ChanEvent*>::const_iterator it2 = it1 + 1;
304  it2 != geEvents_.end(); ++it2) {
305 
306  ChanEvent *chan2 = *it2;
307 
308  double gEnergy2 = chan2->GetCalEnergy();
309  if (gEnergy2 < gammaThreshold_)
310  continue;
311 
312  double gTime2 = chan2->GetCorrectedTime();
313  double gg_dtime = (gTime2 - gTime) * clockInSeconds;
314  if (abs(gg_dtime) > gammaGammaLimit_)
315  continue;
316 
317  symplot(neutron::DD_ENERGY, gEnergy, gEnergy2);
318  if (hasBeta && GoodGammaBeta(gb_dtime)) {
319  symplot(neutron::betaGated::DD_ENERGY,
320  gEnergy, gEnergy2);
321  }
322  }
323  }
324

326 
327  for (unsigned i = 0; i < nEvents; ++i) {
328  double gEnergy = tas_[i].energy;
329  double gTime = tas_[i].time;
330  if (gEnergy < gammaThreshold_)
331  continue;
332 
333  plot(neutron::D_ADD_ENERGY_TOTAL, gEnergy);
334  double gb_dtime = numeric_limits<double>::max();
335  if (hasBeta) {
336  EventData bestBeta = BestBetaForGamma(gTime);
337  gb_dtime = (gTime - bestBeta.time) * clockInSeconds;
338  }
339 
340  if (hasBeta && GoodGammaBeta(gb_dtime)) {
341  plot(neutron::betaGated::D_ADD_ENERGY_TOTAL, gEnergy);
342  }
343 
344  if (neutron_count > 1) {
345  plot(multiNeutron::D_ADD_ENERGY_TOTAL, gEnergy);
346  if (hasBeta && GoodGammaBeta(gb_dtime)) {
347  plot(multiNeutron::betaGated::D_ADD_ENERGY_TOTAL, gEnergy);
348  }
349  }
350  }
351

353  for (unsigned int det = 0; det < numClovers; ++det) {
354  double gEnergy = addbackEvents_[det][ev].energy;
355  double gTime = addbackEvents_[det][ev].time;
356  double decayTime = (gTime - cycleTime) * clockInSeconds;
357  if (gEnergy < gammaThreshold_)
358  continue;
359 
360  plot(neutron::D_ADD_ENERGY, gEnergy);
361  granploty(neutron::DD_ADD_ENERGY__TIMEX, gEnergy,
362  decayTime, timeResolution);
363 
364  double gb_dtime = numeric_limits<double>::max();
365  if (hasBeta) {
366  EventData bestBeta = BestBetaForGamma(gTime);
367  gb_dtime = (gTime - bestBeta.time) * clockInSeconds;
368  }
369

371  plot(neutron::betaGated::D_ADD_ENERGY, gEnergy);
372  granploty(neutron::betaGated::DD_ADD_ENERGY__TIMEX,
373  gEnergy, decayTime, timeResolution);
374  }
375  if (neutron_count > 1) {
376  plot(multiNeutron::D_ADD_ENERGY, gEnergy);
377  granploty(multiNeutron::DD_ADD_ENERGY__TIMEX,
378  gEnergy, decayTime, timeResolution);
379  if (hasBeta && GoodGammaBeta(gb_dtime)) {
380  plot(multiNeutron::betaGated::D_ADD_ENERGY, gEnergy);
381  granploty(multiNeutron::betaGated::DD_ADD_ENERGY__TIMEX,
382  gEnergy, decayTime, timeResolution);
383  }
384  }
385 
386  for (unsigned int det2 = det + 1; det2 < numClovers; ++det2) {
387  double gEnergy2 = addbackEvents_[det2][ev].energy;
388  if (gEnergy2 < gammaThreshold_)
389  continue;
390 
391  double gTime2 = addbackEvents_[det2][ev].time;
392  double gg_dtime = (gTime2 - gTime) * clockInSeconds;
393  if (abs(gg_dtime) > gammaGammaLimit_)
394  continue;
395 
396  symplot(neutron::DD_ADD_ENERGY, gEnergy, gEnergy2);
397  if (hasBeta && GoodGammaBeta(gb_dtime)) {
398  symplot(neutron::betaGated::DD_ADD_ENERGY, gEnergy,
399  gEnergy2);
400  }
401  } // iteration over other clovers
402  } // iteration over clovers
403  } // iteration over events
404 
405  EndProcess();
406  return true;
