87 bool DssdProcessor::Process(RawEvent &event) {
88  using namespace dammIds::dssd;
89 
90  if (!EventProcessor::Process(event))
91  return false;
92  // first time through, grab the according detector summaries
93  if (frontSummary == NULL)
94  frontSummary = event.GetSummary("dssd_front");
95  if (backSummary == NULL)
96  backSummary = event.GetSummary("dssd_back");
97  //Removed this until it can be updated with the TreeCorrelator
98  //static Correlator &corr = event.GetCorrelator();
99 
100  int frontPos = INT_MAX, backPos = INT_MAX;
101  double frontEnergy, backEnergy, frontTime = 0.;
102 
103  bool hasFront = (frontSummary->GetMult() > 0);
104  bool hasBack = (backSummary->GetMult() > 0);
105  bool hasMcp = (event.GetSummary("mcp")->GetMult() > 0);
106 
107  if (hasFront) {
108  const ChanEvent *ch = frontSummary->GetMaxEvent();
109 
110  frontPos = ch->GetChanID().GetLocation();
111  frontEnergy = ch->GetCalEnergy();
112  frontTime = ch->GetTime();
113  } else frontEnergy = 0.;
114  if (hasBack) {
115  const ChanEvent *ch = backSummary->GetMaxEvent();
116 
117  backPos = ch->GetChanID().GetLocation();
118  backEnergy = ch->GetCalEnergy();
119  } else backEnergy = 0.;
120 
121  // decide whether this is an implant or a decay
122  EventInfo corEvent;
123  corEvent.time = frontTime;
124 
125  if (hasFront && hasBack) {
126  if (frontEnergy > cutoffEnergy && backEnergy > cutoffEnergy && hasMcp) {
127  corEvent.type = EventInfo::IMPLANT_EVENT;
128  } else if (frontEnergy <= cutoffEnergy && backEnergy <= cutoffEnergy &&
129  !hasMcp) {
130  corEvent.type = EventInfo::DECAY_EVENT;
131  } else {
132  corEvent.type = EventInfo::UNKNOWN_EVENT;
133  }
134  corr.Correlate(corEvent, frontPos, backPos);
135  } else if (hasFront) {
136  if (frontEnergy > cutoffEnergy) {
137  corEvent.type = EventInfo::IMPLANT_EVENT;
138  } else corEvent.type = EventInfo::DECAY_EVENT;
139 
140  } else if (hasBack) {
141  if (backEnergy > cutoffEnergy) {
142  corEvent.type = EventInfo::IMPLANT_EVENT;
143  } else corEvent.type = EventInfo::DECAY_EVENT;
144  } else corEvent.type = EventInfo::UNKNOWN_EVENT;
145 
146  // plot stuff
147  if (corEvent.type == EventInfo::IMPLANT_EVENT) {
148  if (hasFront)
149  plot(DD_IMPLANT_FRONT_ENERGY__POSITION, frontEnergy, frontPos);
150  if (hasBack)
151  plot(DD_IMPLANT_BACK_ENERGY__POSITION, backEnergy, backPos);
152  if (hasFront && hasBack)
153  plot(DD_IMPLANT_POSITION, backPos, frontPos);
154  } else if (corEvent.type == EventInfo::DECAY_EVENT) {
155  if (hasFront)
156  plot(DD_DECAY_FRONT_ENERGY__POSITION, frontEnergy, frontPos);
157  if (hasBack)
158  plot(DD_DECAY_BACK_ENERGY__POSITION, backEnergy, backPos);
159  if (hasFront && hasBack)
160  plot(DD_DECAY_POSITION, backPos, frontPos);
161  if (corr.GetCondition() == Correlator::VALID_DECAY) {
162  const unsigned int NumGranularities = 8;
163  // time resolution in seconds per bin
164  const double timeResolution[NumGranularities] =
165  {10e-9, 100e-9, 400e-9, 1e-6, 100e-6, 1e-3, 10e-3, 100e-3};
166 
167  for (unsigned int i = 0; i < NumGranularities; i++) {
168  int timeBin = int(corr.GetDecayTime() *
169  Globals::get()->clockInSeconds() / timeResolution[i]);
170 
171  plot(DD_ENERGY__DECAY_TIME_GRANX + i, frontEnergy, timeBin);
172  }
173  }
174  }
175 
176  EndProcess(); // update the processing time
177  return true;
