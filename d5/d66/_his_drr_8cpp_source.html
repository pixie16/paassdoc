80 HisDrr::HisDrr(const string &drr, const string &his, const string &input) {

82  if ( sizeof(unsigned short) != 2 || sizeof(unsigned int) != 4 ) {
83  stringstream err;
84  err << "HisDrr:4: This program is intended to run with 'unsigned short' size 2 bytes and"
85  << " and 'unsigned int' size 4 bytes. Your machine uses " << sizeof(unsigned short)
86  << " and " << sizeof(unsigned int) << " respectively." << endl;

91  ifstream fileInput(input.c_str());
92  if (!fileInput.good()) {
93  stringstream err;
94  err << "HisDrr:5: Could not open file " << input;
95  string msg = err.str();
96  fileInput.close();
97  throw IOError(msg);
98  }
99 
100  //This part fills in drrData vector using definition found in input text file
101  vector <SimpleDrrBlock> drrData;
102  if (fileInput.good()) {
103  string line;
104  while (getline(fileInput, line) ) {
105  istringstream iss;
106  iss.str(line);
107  if (line[0] != '#') {
108  string token;
109  SimpleDrrBlock block;
110  iss >> block.hisID >> block.halfWords >> block.scaled[0] >> block.scaled[1];
111  if ((block.halfWords != 1)&&(block.halfWords != 2)) {
112  stringstream err;
113  err << "HisDrr:6: Only 2 or 4 bytes long histograms supported";
114  string msg = err.str();
115  fileInput.close();
116  throw GenError(msg);
117  }
118  while (iss) {
119  string titlePart;
120  iss >> titlePart;
121  block.title += titlePart + " ";
122  }
123  if (block.title.size() == 1)
124  block.title = "Empty title ";
125  drrData.push_back(block);
126  iss.clear();
127  }
128  }
129  fileInput.close();
130  }
131 
132  drrFile = new fstream(drr.c_str(), fstream::binary | fstream::in | fstream::out | fstream::trunc);
133  hisFile = new fstream(his.c_str(), fstream::binary | fstream::in | fstream::out | fstream::trunc);
134 
135  if (!drrFile->good()) {
136  stringstream err;
137  err << "HisDrr:7: Could not create file " << drr;
138  string msg = err.str();
139  drrFile->close();
140  throw IOError(msg);
141  }
142 
143  if (!hisFile->good()) {
144  stringstream err;
145  err << "HisDrr:8: Could not create file " << his;
146  string msg = err.str();
147  hisFile->close();
148  throw IOError(msg);
149  }
150 
151  // Using information from drrData drr header is created
152  DrrHeader head;
153  int totLength;
154  for (unsigned int i = 0; i < drrData.size(); ++i)
155  totLength += (drrData[i].scaled[0]+drrData[i].scaled[1])*drrData[i].halfWords;
156  // Magic words (whatever they do...)
157  string initial = "HHIRFDIR0001";
158  for (unsigned int i = 0; i < initial.size(); ++i) {
159  head.initial[i] = initial[i];
160  }
161  head.nHis = drrData.size();
162  head.nHWords = totLength;
163  time_t clock = time(NULL);
164  tm *date = localtime(&clock);
165  head.date[0]=0;
166  head.date[1]=date->tm_year+1900;
167  head.date[2]=date->tm_mon;
168  head.date[3]=date->tm_mday;
169  head.date[4]=date->tm_hour;
170  head.date[5]=date->tm_min;
171  char description[40] = {0};
172  for (int i = 0; i < 40; ++i)
173  head.description[i] = description[i];
174  // Header is followed by 44 empty bytes to reach 128 bytes long block
175  drrFile->write((char *)&head, sizeof(head));
176  char garbage[44] = {0};
177  drrFile->write(garbage, 44);
178 
179  DrrHisRecord record;
180 
181  // This part is creating and writing records for each histogram
182  // At the same time filling in the his file with '0' values
183  int offset = 0;
184  for (unsigned int i = 0; i < drrData.size(); ++i) {
185  int dim = 0;
186  for (int j = 0; j < 2; ++j)
187  if (drrData[i].scaled[j] > 0)
188  dim++;
189  record.hisDim = dim;
190  record.halfWords = drrData[i].halfWords;
191 
192  for (int j = 0; j < 4; ++j)
193  record.params[j] = 0;
194 
195  record.raw[0] = drrData[i].scaled[0];
196  record.raw[1] = drrData[i].scaled[1];
197  for (int j = 2; j < 4; ++j)
198  record.raw[j] = 0;
199 
200  record.scaled[0] = drrData[i].scaled[0];
201  record.scaled[1] = drrData[i].scaled[1];
202  for (int j = 2; j < 4; ++j)
203  record.scaled[j] = 0;
204 
205  for (int j = 0; j < 4; ++j)
206  record.minc[j] = 0;
207 
208  record.maxc[0] = drrData[i].scaled[0]-1;
209  if (drrData[i].scaled[1] > 0)
210  record.maxc[1] = drrData[i].scaled[1]-1;
211  else
212  record.maxc[1] = 0;
213  for (int j = 2; j < 4; ++j)
214  record.maxc[j] = 0;
215 
216  record.offset = offset;
217  char label[12] = {0};
218  for (int j = 0; j < 12; ++j) {
219  record.xlabel[j] = label[j];
220  record.ylabel[j] = label[j];
221  }
222  for (int j = 0; j < 4; ++j)
223  record.calcon[j] = 0;
224  unsigned int k = 0;
225  while ( (k < 39)&&(k < drrData[i].title.size() - 1) ){
226  record.title[k] = drrData[i].title[k];
227  k++;
228  }
229  //It is very important to fill unused part of title with 0
230  //As well as remove end of string character from the last position
231  for (; k < 40; k++)
232  record.title[k] = 0;
233 
234  drrFile->write((char *)&record, sizeof(record));
235 
236  unsigned size = 0;
237  if (dim == 1)
238  size = drrData[i].scaled[0];
239  if (dim == 2)
240  size = drrData[i].scaled[0]*drrData[i].scaled[1];
241 
242  if (drrData[i].halfWords == 1) {
243  unsigned short *his = new unsigned short[size]();
244  hisFile->write((char *)his, size*sizeof(unsigned short));
245  offset += size*drrData[i].halfWords;
246  delete []his;
247  }
248  if (drrData[i].halfWords == 2) {
249  unsigned int *his = new unsigned int[size]();
250  hisFile->write((char *)his, size*sizeof(unsigned int));
251  offset += size*drrData[i].halfWords;
252  delete []his;
253  }
254 
255  }
256 
257  //At the end of file we put a list of histograms in 128 bytes long blocks
258  //They are build of 32 records of 4 bytes (int) long histogram Id's
259  for (unsigned int i = 0; i < drrData.size()/32 + 1; ++i) {
260  int hisList[32] = {0};
261  unsigned int j = 0;
262  while ((j < drrData.size())&&(j < 32)) {
263  hisList[j%32] = drrData[i*32+j].hisID;
264  ++j;
265  }
266  drrFile->write((char *)hisList, 128);
267  }
268  //Testing messages
269  // cout << "# Created files " << drr << " and " << his << " with following histograms: " << endl;
270  // for (unsigned int i = 0; i < drrData.size(); ++i) {
271  // cout << "# " << drrData[i].hisID << " " << drrData[i].halfWords << " " << drrData[i].scaled[0] << " " << drrData[i].scaled[1] << " \"" << drrData[i].title << "\"" << endl;
272  // }
273 
274  // And now constructors loads freshly created files
275  // so it follows a logic scheme of constructor(drr, his)
276  loadDrr();

290 void HisDrr::loadDrr() {
291  // Block of data 128 lenght stored in union
292  drrBlock block;
293 
294  if (drrFile->good()) {
295  drrFile->seekg(0, ios::beg);
296  readBlock(&block);
297  }
298 
299  // Header contains number of histograms in a file
300  int nHis = block.header.nHis;
301 
302  // We read as many blocks of data as the number of
303  // histogram
304  //
305  // Inside a loop there is a jump made.
306  // We have to jump inside the drr file to get to
307  // the histogram IDs which are at the end of file.
308  // Jump lenght is equal to:
309  // number of histograms plus one (for header) plus
310  // current histogram index (position in file, not id)
311 
312  int hId;
313  DrrHisRecordExtended drrRecExt;
314  hisList.reserve(nHis);
315  for (int i = 0; i < nHis; ++i) {
316  if (drrFile->good()) {
317  readBlock(&block);
318  int currentPos = drrFile->tellg();
319  //jump for ID
320  drrFile->seekg( (nHis+1)*sizeof(block)+i*sizeof(hId) );
321  //check if we are still in the file
322  if (drrFile->good())
323  drrFile->read((char*)&hId, sizeof(hId));
324  else {
325  stringstream err;
326  err << "HisDrr:10: Error reading drr file at " << drrFile->tellg();
327  string msg = err.str();
328  throw IOError(msg);
329  }
330  //...and jump back
331  drrFile->seekg(currentPos);
332 
333  drrRecExt.hisID = hId;
334  drrRecExt = block.record;
335  hisList.push_back(drrRecExt);
336  }
337  else {
338  stringstream err;
339  err << "HisDrr:11: Error reading drr file at " << drrFile->tellg();
340  string msg = err.str();
341  throw IOError(msg);
342  }
343  }

347 void HisDrr::getHistogram(vector<unsigned int> &rtn, int id) {
348  // First we search if histogram id exists
349  int index = -1;
350  for (unsigned int i = 0; i < hisList.size(); ++i)
351  if (hisList[i].hisID == id) {
352  index = i;
353  break;
354  }
355  if (index < 0) {
356  stringstream err;
357  err << "HisDrr:12: Could not find spectrum id = " << id << " in drr file";
358  string msg = err.str();
359  throw GenError(msg);
360  }
361 
362  vector<unsigned int> r;
363  if (hisFile->good()) {
364  // Set position of pointer in file to the beginning
365  hisFile->seekg(0, ios::beg);
366  // We jump to location specified by offset (given in units of 2 bytes)
367  hisFile->seekg(hisList[index].offset*2);
368  // Lenght of data is equal to product of all histogram dimensions lengths
369  unsigned int length = 1;
370  for (int i = 0; i < hisList[index].hisDim; ++i)
371  length = length * hisList[index].scaled[i];
372 
373  // Check if data exceedes size of unsigned int
374  if ((unsigned short)(hisList[index].halfWords*2) > sizeof(unsigned int) ) {
375  stringstream err;
376  err << "HisDrr:13: Histograms with channel size " << hisList[index].halfWords*2
377  << " bytes long are not supported ";
378  string msg = err.str();
379  throw GenError(msg);
380  }
381 
382  // Since we know now the length of data, we can reserve space
383  // for vector avoiding unecessary data movement in memory
384  r.reserve(length);
385  // We read histogram channels and push them to vector
386  // Each channel has a length of equal to halfWords parameter
387  // which is given in 2 bytes units
388  // We cast this value on unsigned int, which is zeroed first
389  // It is a safe cast then.
390  for (unsigned int i = 0; i < length; ++i) {
391  unsigned int u = 0;
392  hisFile->read((char*)&u, hisList[index].halfWords*2);
393  r.push_back(u);
394  }
395  }
396  // Return by value version
397  //return r;
398 
399  // Return by reference version
400  rtn.swap(r);
401 }
