386 bool GeProcessor::PreProcess(RawEvent &event) {
387  if (!EventProcessor::PreProcess(event))
388  return false;
389 
390  geEvents_.clear();
391  for (unsigned i = 0; i < numClovers; ++i)
392  addbackEvents_[i].clear();
393  tas_.clear();
394 
395  static const vector<ChanEvent*> &highEvents =
396  event.GetSummary("ge:clover_high", true)->GetList();
397  static const vector<ChanEvent*> &lowEvents =
398  event.GetSummary("ge:clover_low", true)->GetList();
399 
403  for (vector<ChanEvent*>::const_iterator itHigh = highEvents.begin();
404  itHigh != highEvents.end(); itHigh++) {
405  int location = (*itHigh)->GetChanID().GetLocation();
406  if ( (*itHigh)->IsSaturated() || (*itHigh)->IsPileup() )
407  continue;
408 
409  vector <ChanEvent*>::const_iterator itLow = lowEvents.begin();
410  for (; itLow != lowEvents.end(); itLow++) {
411  if ( (*itLow)->GetChanID().GetLocation() == location ) {
412  break;
413  }
414  }
415  if ( itLow != lowEvents.end() ) {
416  double ratio = (*itHigh)->GetEnergy() / (*itLow)->GetEnergy();
417  if (ratio < lowRatio_ || ratio > highRatio_)
418  continue;
419  }
420  geEvents_.push_back(*itHigh);
421  }
422 
423  // now we sort the germanium events according to their corrected time
424  sort(geEvents_.begin(), geEvents_.end(), CompareCorrectedTime);
425 
435  double refTime = -2.0 * subEventWindow_;
436 
437  for (vector<ChanEvent*>::iterator it = geEvents_.begin(); it != geEvents_.end(); it++) {
438  ChanEvent *chan = *it;
439  double energy = chan->GetCalEnergy();
440  double time = chan->GetCorrectedTime();
441  int clover = leafToClover[chan->GetChanID().GetLocation()];
442 
447  if (energy < gammaThreshold_)
448  continue;
449 
450  // entries in map are sorted by time
451  // if event time is outside of subEventWindow, we start new
452  // events for all clovers and "tas"
453  double dtime = abs(time - refTime) * Globals::get()->clockInSeconds();
454  if (dtime > subEventWindow_) {
455  for (unsigned i = 0; i < numClovers; ++i) {
456  addbackEvents_[i].push_back(AddBackEvent());
457  }
458  tas_.push_back(AddBackEvent());
459  }
460  // Total addback energy
461  addbackEvents_[clover].back().energy += energy;
462  // We store latest time only
463  addbackEvents_[clover].back().time = time;
464  addbackEvents_[clover].back().multiplicity += 1;
465  tas_.back().energy += energy;
466  tas_.back().time = time;
467  tas_.back().multiplicity += 1;
468  refTime = time;
469  }
470 
471  return true;
472 }
