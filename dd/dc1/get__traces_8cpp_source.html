86 double TraceGrabber::FitTau(const unsigned short* trace, size_t b0, size_t b1, size_t x0, size_t x1)
87 {
88  double baseline = 0;
89  for (unsigned i = b0; i < b1; ++i)
90  baseline += trace[i];
91  baseline = baseline / double(b1 - b0);
92 
93  //calculate logarithm
94  vector<double> logtrace;
95  for (size_t i = x0; i < x1; ++i) {
96  double val = trace[i] - baseline;
97  if (val > 0)
98  logtrace.push_back( log(val) );
99  else
100  logtrace.push_back( -6.66E-6 ); //such an evil number!
101  }
102 
103  // fit a line
104  double S = (x1 - x0);
105  double Sx = ( pow(x1,2) - x1 - pow(x0,2) + x0) / 2.0;
106  double Sxx = ( 2.0 * pow(x1,3) - 3.0 * pow(x1,2) + x1
107  -2.0 * pow(x0,3) + 3.0 * pow(x0,2) - x0) / 6.0;
108  double Sy = 0;
109  double Sxy = 0;
110 
111  unsigned sz = logtrace.size();
112  for (unsigned i = 0; i < sz; i++) {
113  Sy += logtrace[i];
114  Sxy += logtrace[i] * i;
115  }
116  double D = S * Sxx - pow(Sx,2);
117  double a1 = (S * Sxy - Sx * Sy) / D;
118  //double a0 = (Sxx * Sy - Sx * Sxy) / D;
119 
120  if ( a1 != 0.0)
121  return (-1.0 / a1);
122  else
123  return -6.66e6;
124 }

126 bool TraceGrabber::operator()(PixieFunctionParms<> &par)
127 {
128  /* Grab traces the first time around */
129  static size_t loopCount = loopLength;
130  // Reads traces from whole module
131  // Should be run only once per 'loop' (when all individual channels
132  // are processed)
133  if (loopCount >= loopLength) {
134  par.pif.AcquireTraces(par.mod);
135  loopCount = 0;
136  }
137  // If channel ready skip the rest
138  if ( ready[par.ch] )
139  return true;
140  loopCount++;
141 
142  const size_t size = PixieInterface::GetTraceLength();
143  unsigned short *trace = new unsigned short[size];
144  usleep(10);
145  if (par.pif.ReadSglChanTrace(trace, size, par.mod, par.ch)) {
146  if (trigger > 0.0 && attempts[par.ch] < maxTries) {
147  if (attempts[par.ch] == 0 || updateBaselines) {
148  size_t baselineSamples = size / 10;
149  unsigned long sum = 0;
150 
151  for (unsigned i = 0; i < baselineSamples; ++i)
152  sum += trace[i];
153 
154  baseline[par.ch] = float(sum)/float(baselineSamples);
155  if ( attempts[par.ch] == 0 ) {
156  cout << "CH: " << par.ch << " average " << baseline[par.ch];
157  cout << ", trig level = " << baseline[par.ch] + trigger;
158  if (isBidirectional)
159  cout << " or " << baseline[par.ch] - trigger;
160  if (updateBaselines)
161  cout << ", updating continuously";
162  cout << endl;
163  }
164  }
165 
166  attempts[par.ch]++;
167 
168  bool trig = false;
169  int triggerHigh = baseline[par.ch] + trigger;
170  int triggerLow = baseline[par.ch] - trigger;
171 
172  size_t x0;
173  for (unsigned int i = 0; i < size; ++i) {
174  if (trace[i] > triggerHigh ||
175  (isBidirectional && trace[i] < triggerLow) ) {
176  trig = true;
177  x0 = i;
178  break;
179  }
180  }
181  if (trig) {
182  StoreData(trace, size, par.ch);
183  cout << "Channel " << par.ch << " ready " << " trig " << trace[x0] << " at " << x0 << endl;
184  if (doFit) {
185  size_t x1 = 0;
186  for (size_t i = size - 1; i > x0; --i) {
187  if (trace[i] > triggerHigh - trigger/2 ||
188  (isBidirectional && trace[i] < triggerLow + trigger / 2) ) {
189  x1 = i;
190  break;
191  }
192  }
193  // Arbitrary 10 samples between x0 and x1
194  if (x1 > x0 + 10) {
195  size_t b0, b1;
196 
197  if (x0 > 30) {
198  b0 = 0;
199  b1 = 20;
200  } else {
201  b0 = 8171;
202  b1 = 8191;
203  }
204  // x0 + 5 in order to get rid of growin part
205  // might be not enough
206  double tau = FitTau(trace, b0, b1, x0+5, x1);
207 
208  cout << "Tau = " << tau << endl;
209  cout << " x0 = " << x0 << ", x1 = " << x1 << endl;
210  cout << " b0 = " << x0 << ", b1 = " << x1 << endl;
211  } else {
212  // not enough samples
213  cout << "Could not find tau " << endl;
214  }
215  }
216  }
217  } else if (attempts[par.ch] >= maxTries) {
218  attempts[par.ch]++;
219  StoreData(trace, size, par.ch);
220  cout << "Channel " << par.ch << " exceeded max number of attempts " << endl;
221  } else if (trigger == 0.0 ) {
222  StoreData(trace, size, par.ch);
223  cout << "Channel " << par.ch << " ready, triggerless" << endl;
224  }
225  delete[] trace;
226  return true;
227  } else {
228  // Did not successfully read trace
229  delete[] trace;
230  return false;
231  }
232 }
