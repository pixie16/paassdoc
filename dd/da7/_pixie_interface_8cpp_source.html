155  if (adcResLocBegin == string::npos) adcResLocBegin = 0;

171  size_t mspsLocEnd = value.find('m');
172  if (mspsLocEnd != string::npos) {

174  if (mspsLocBegin == string::npos) mspsLocBegin = 0;
175  else mspsLocBegin++;
176  mspsStr = value.substr(mspsLocBegin, mspsLocEnd - mspsLocBegin);
177  }

179  if (mspsStr.empty()) {

183  try { msps = std::stoi(mspsStr); }

186  }
187  }

369 bool PixieInterface::Boot(int mode, bool useWorkingSetFile)
370 {
371  string &setFile = useWorkingSetFile ?
372  configStrings["global"]["DspWorkingSetFile"] : configStrings["global"]["DspSetFile"];
373 
374  LeaderPrint("Boot Configuration");
375 
376  //Loop through each module and determine its type.
377  //We also check if the modules are all the same. If not we set multiConf to true.
378  bool multiConf = false;
379  std::vector< std::string > moduleTypes;
380  for (int mod = 0; mod < numberCards; mod++) {
381  unsigned short rev, adcBits, adcMsps;
382  unsigned int serNum;
383  GetModuleInfo(mod, &rev, &serNum, &adcBits, &adcMsps);
384 
385  stringstream moduleType;
386  moduleType << adcBits << "b" << adcMsps << "m";
387  moduleType << "-rev" << (char)(97 + rev - 10 );
388 
389  if (mod > 0 && moduleType.str() != moduleTypes.back()) multiConf = true;
390  moduleTypes.push_back(moduleType.str());
391  }
392 
393  bool goodBoot = true;
394 
395  if (multiConf) {
396  // must proceed through boot module by module
397 
398  //Break the leader print for the boot configuration status.
399  cout << InfoStr("[MULTI]") << "\n";
400 
401  //Check that all module types are valid.
402  bool error = false;
403  for (int mod = 0; mod < numberCards; mod++) {
404  if (configStrings.find(moduleTypes.at(mod)) == configStrings.end()) {
405  std::cout << ErrorStr() << " Configuration not defined for type " << moduleTypes.at(mod) << " (mod " << mod << ")\n";
406  error = true;
407  }
408  }
409  if (error) return false;
410 
411  for (int i=0; i < numberCards; i++) {
412  retval = Pixie16BootModule(&configStrings[moduleTypes.at(i)]["ComFpgaFile"][0],
413  &configStrings[moduleTypes.at(i)]["SpFpgaFile"][0],
414  &configStrings[moduleTypes.at(i)]["TrigFpgaFile"][0],
415  &configStrings[moduleTypes.at(i)]["DspConfFile"][0],
416  &setFile[0],
417  &configStrings[moduleTypes.at(i)]["DspVarFile"][0],
418  i, mode);
419 
420  stringstream leader;
421  leader << "Booting Pixie (" << moduleTypes.at(i) << ") Module " << i;
422  LeaderPrint(leader.str());
423  goodBoot = (goodBoot && !CheckError(true));
424  }
425  } else {
426  //Break the leader print for the boot configuration status.
427  cout << InfoStr("[SINGLE]") << "\n";
428 
429  //Determine if we need to use "default" type.
430  string moduleType = moduleTypes.front();
431  if (configStrings.find(moduleType) == configStrings.end()) {
432  moduleType = "default";
433  if (configStrings.find(moduleType) == configStrings.end()) {
434  std::cout << ErrorStr() << " Config not defined for type " << moduleTypes.back() << "\n";
435  return false;
436  }
437  }
438  //std::cout << "Booting all modules as type " << InfoStr(moduleType) << "\n";
439 
440  // boot all at once
441  retval = Pixie16BootModule(&configStrings[moduleType]["ComFpgaFile"][0],
442  &configStrings[moduleType]["SpFpgaFile"][0],
443  &configStrings[moduleType]["TrigFpgaFile"][0],
444  &configStrings[moduleType]["DspConfFile"][0],
445  &setFile[0],
446  &configStrings[moduleType]["DspVarFile"][0],
447  numberCards, mode);
448 
449  stringstream leader;
450  leader << "Booting Pixie (" << moduleType << ")";
451  LeaderPrint(leader.str());
452  goodBoot = !CheckError(true);
453  }
454 
455  cout << " Used set file: " << InfoStr(setFile) << endl;
456 
457  LeaderPrint("Checking SlotIDs");
458 
459  bool hadError = false;
460  bool updated = false;
461 
462  word_t val;
463 
464  for (int i=0; i < numberCards; i++) {
465  if (!ReadSglModPar("SlotID", val, i))
466  hadError = true;
467  if (val != slotMap[i]) {
468  updated = true;
469  if (!WriteSglModPar("SlotID", slotMap[i], i))
470  hadError = true;
471  }
472  }
473 
474  if (hadError)
475  cout << ErrorStr() << endl;
476  else if (updated)
477  cout << WarningStr("[UPDATED]") << endl;
478  else
479  cout << OkayStr() << endl;
480 
481  return goodBoot && !hadError;
482 }
