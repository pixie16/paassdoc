157  bool tapeMove = !(TreeCorrelator::get()->place("Cycle")->status());

161 
162  for (vector<ChanEvent*>::const_iterator it = scintBetaEvents.begin();
163  it != scintBetaEvents.end(); it++) {
164  double energy = (*it)->GetCalEnergy();
165  int energyBin = int(energy / energyContraction_);
166  double time = (*it)->GetTime();
167 
168  //Skip the energy-time spectra for zero energy events
169  if (energyBin < 1)
170  continue;
171 
172  double decayTime = (time - cycleTime) * clockInSeconds;
173  int decayTimeBin = int(decayTime / timeSpectraTimeResolution);
174 
175  EventData bestGamma = BestGammaForBeta(time);
176  double gb_dtime = (time - bestGamma.time) * clockInSeconds;
177 
178  if (GoodGammaBeta(gb_dtime)) {
179  plot(neutron::D_ENERGY_BETA_GAMMA_GATED, energyBin);
180  plot(neutron::DD_ENERGY_BETA__GAMMA, energyBin, bestGamma.energy);
181  }
