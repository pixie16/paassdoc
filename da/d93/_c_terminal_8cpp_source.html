740 void Terminal::TabComplete(const std::string &input_, const std::vector<std::string> &possibilities_) {
741  if(input_.empty() || possibilities_.empty()) return;
742 
743  std::vector<std::string> matches;
744  std::vector<std::string>::iterator it;
745 
746  size_t start = input_.find_last_of(" ");
747  if(start == std::string::npos)
748  start = 0;
749  else
750  start++;
751 
752  size_t stop = input_.length()-1; // Strip off the trailing '\t'.
753 
754  //Get the string to complete
755  std::string strToComplete = input_.substr(start, stop-start);
756 
757 
758  for (auto it=possibilities_.begin(); it!=possibilities_.end();++it) {
759  if (it->empty())
760  continue;
761  if (it->find(strToComplete) == 0)
762  matches.push_back(it->substr(strToComplete.length()));
763  }
764 
765  //No tab complete matches so we do nothing.
766  if (matches.size() == 0) {
767  return;
768  }
769 
770  //A unique match so we extend the command with completed text
771  else if (matches.size() == 1) {
772  if (matches.at(0).find("/") == std::string::npos && (unsigned int) (cursX - offset) == cmd.length()) {
773  matches.at(0).append(" ");
774  }
775  cmd.insert(cursX - offset, matches.at(0).c_str());
776  waddstr(input_window, cmd.substr(cursX - offset).c_str());
777  cursX += matches.at(0).length();
778  }
779  else {
780  //Fill out the matching part
781  std::string commonStr = matches.at(0);
782  for (auto it=matches.begin()+1;it!=matches.end() && !commonStr.empty();++it) {
783  while (!commonStr.empty()) {
784  if ((*it).find(commonStr) == 0) break;
785  commonStr.erase(commonStr.length() - 1);
786  }
787  }
788  if (!commonStr.empty()) {
789  cmd.insert(cursX - offset, commonStr.c_str());
790  waddstr(input_window, cmd.substr(cursX - offset).c_str());
791  cursX += commonStr.length();
792  }
793  //Display the options
794  else if (tabCount > 1) {
795  //Compute the header position
796  size_t headerPos = cmd.find_last_of(" /",cursX - offset - 1);
797  std::string header;
798  if (headerPos == std::string::npos)
799  header = cmd;
800  else
801  header = cmd.substr(headerPos+1,cursX - offset - 1 - headerPos);

1121  args.clear();
1122 
1123  //Locate the first non delimiter space.
1124  size_t strStart = str.find_first_not_of(delimiter);
1125  //In case of a line with only delimiters we return 0.
1126  if (strStart == std::string::npos) return 0;
1127 
1128  //Locate the last character that is not a delimiter.
1129  size_t strStop = str.find_last_not_of(delimiter) + 1;
1130 
1131  //We loop over the string searching for the delimiter keeping track of where
1132  // we found the current delimiter and the previous one.
1133  size_t pos = strStart;
1134  size_t lastPos = strStart;
1135  while ((pos = str.find(delimiter, lastPos)) != std::string::npos) {
1136  //Store the substring from the last non delimiter to the current delimiter.
1137  args.push_back(str.substr(lastPos, pos - lastPos));
1138 
1139  //We update the last position looking for the first character that is not
1140  // a delimiter.
1141  lastPos = str.find_first_not_of(delimiter, pos+1);
1142  }
1143 
1144  //Store the last string.
1145  if(lastPos != std::string::npos)
1146  args.push_back(str.substr(lastPos, strStop - lastPos));
1147 
1148  return args.size();
