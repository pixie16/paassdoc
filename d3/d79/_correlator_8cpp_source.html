195 void Correlator::Correlate(EventInfo &event, unsigned int fch,
196  unsigned int bch) {
197  using namespace dammIds::correlator;
198  if(fch < 0 || fch >= arraySize || bch < 0 || bch >= arraySize) {
199  plot(D_CONDITION, INVALID_LOCATION);
200  return;
201  }
202  CorrelationList &theList = decaylist[fch][bch];
203  double lastTime = NAN;
204  double clockInSeconds = Globals::get()->clockInSeconds();
205  switch(event.type) {
206  case EventInfo::IMPLANT_EVENT:
207  if(theList.IsFlagged()) {
208  PrintDecayList(fch, bch);
209  }
210  lastTime = GetImplantTime(fch, bch);
211  theList.clear();
212  condition = VALID_IMPLANT;
213  if(lastImplant != NULL) {
214  double dt = event.time - lastImplant->time;
215  plot(D_TIME_BW_ALL_IMPLANTS, dt * clockInSeconds / 1e-6);
216  }
217  if(!isnan(lastTime)) {
218  condition = BACK_TO_BACK_IMPLANT;
219  event.dtime = event.time - lastTime;
220  plot(D_TIME_BW_IMPLANTS, event.dtime * clockInSeconds / 100e-3);
221  }
222  else {
223  event.dtime = INFINITY;
224  }
225  event.generation = 0;
226  theList.push_back(event);
227  lastImplant = &theList.back();
228  break;
229  default:
230  if(theList.empty()) {
231  break;
232  }
233  if(isnan(theList.GetImplantTime())) {
234  cout << "No implant time for decay list" << endl;
235  break;
236  }
237  if(event.type == EventInfo::UNKNOWN_EVENT) {
238  condition = UNKNOWN_CONDITION;
239  }
240  else {
241  condition = VALID_DECAY;
242  }
243  condition = VALID_DECAY; // tmp -- DTM
244  lastTime = theList.back().time;
245  double dt = event.time - theList.GetImplantTime();
246  if(dt < 0) {
247  if(dt < -5e11 && event.time < 1e9) {
248  cout << "Decay following pixie clock reset, clearing decay lists!" << endl;
249  cout << " Event time: " << event.time
250  << "\n Implant time: " << theList.GetImplantTime()
251  << "\n DT: " << dt << endl;
252  // PIXIE's clock has most likely been zeroed due to a file marker
253  // no chance of doing correlations
254  for(unsigned int i=0; i < arraySize; i++) {
255  for(unsigned int j=0; j < arraySize; j++) {
256  if(IsFlagged(i,j)) {
257  PrintDecayList(i, j);
258  }
259  decaylist[i][j].clear();
260  }
261  }
262  }
263  else if(event.type != EventInfo::GAMMA_EVENT) {
264  // since gammas are processed at a different time than everything else
265  cout << "negative correlation time, DECAY: " << event.time
266  << " IMPLANT: " << theList.GetImplantTime()
267  << " DT: " << dt << endl;
268  }
269  event.dtime = NAN;
270  break;
271  } // negative correlation itme
272  if(theList.front().dtime * clockInSeconds >= minImpTime) {
273  if(dt * clockInSeconds < corrTime) {
274  // event.dtime = event.time - lastTime; // (FOR CHAINS)
275  event.dtime = event.time - theList.front().time; // FOR LERIBSS
276  if(event.dtime * clockInSeconds < fastTime && event.dtime > 0) {
277  // event.flagged = true;
278  }
279  }
280  else {
281  // event.dtime = event.time - lastTime; // (FOR CHAINS)
282  event.dtime = event.time - theList.front().time; // FOR LERIBSS
283  condition = DECAY_TOO_LATE;
284  }
285  }
286  else {
287  condition = IMPLANT_TOO_SOON;
288  }
289  if(condition == VALID_DECAY) {
290  event.generation = theList.back().generation + 1;
291  }
292  theList.push_back(event);
293  if(event.energy == 0 && isnan(event.time))
294  cout << " Adding zero decay event " << endl;
295  if(event.flagged)
296  theList.Flag();
297  if(condition == VALID_DECAY) {
298  lastDecay = &theList.back();
299  }
300  else if(condition == DECAY_TOO_LATE) {
301  theList.clear();
302  }
303  break;
304  }
305  plot(D_CONDITION, condition);
306 }
