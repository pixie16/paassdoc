22 // Minimum RISETIME value given by r = 2^(N-1) * (32 ns)
23 // where r is the RISETIME and N is the FILTER_RANGE.
24 #define MIN_TRIGGER_RISETIME 0.016 // us
25 #define MIN_ENERGY_RISETIME 0.032 // us
26 
27 // Minimum FLATTOP value given by f = 2^(N-1) * (48 ns)
28 // where f is the FLATTOP and N is the FILTER_RANGE
29 // except for TRIGGER_FLATTOP which has minumum 0.
30 #define MIN_TRIGGER_FLATTOP 0.000 // us
31 #define MIN_ENERGY_FLATTOP 0.048 // us
32 
33 // Minimum filter step size is given by s = (N+1) * (8 ns)
34 // where s is the step size and N is the FILTER_RANGE.
35 #define MIN_ENERGY_STEP 0.016 // us
36 #define MIN_TRIGGER_STEP 0.008 // us

100  void updateValues(const double &minValue_, const double &minStep_){
101  if(startVal < minValue_)
102  startVal = minValue_;
103 
104  if(stepSize < minStep_)
105  stepSize = minStep_;
106 
107  // Ensure that the values are evenly divisible by the minimum step size.
108  int tempStart = (int)(startVal * 1000);
109  int tempStop = (int)(stopVal * 1000);
110  int tempStep = (int)(stepSize * 1000);
111  int tempMinStep = (int)(minStep_ * 1000);
112 
113  if(debug){
114  std::cout << "start=" << tempStart << " ns, stop=" << tempStop << " ns, step=";
115  std::cout << tempStep << " ns, minStep=" << tempMinStep << " ns\n";
116  }
117 
118  if(tempStart % tempMinStep != 0){ // Go to the next lowest step.
119  tempStart -= tempStart % tempMinStep;
120  startVal = tempStart / 1000.0;
121  }
122  if(tempStop % tempMinStep != 0){ // Go to the next highest step.
123  tempStop += tempStart % tempMinStep;
124  stopVal = tempStop / 1000.0;
125  }
126  if(tempStep % tempMinStep != 0){ // Go to the next lowest step.
127  tempStep -= tempStep % tempMinStep;
128  stepSize = tempStep / 1000.0;
129  }
130 
131  if(debug){
132  std::cout << "start=" << tempStart << " ns, stop=" << tempStop << " ns, step=";
133  std::cout << tempStep << " ns, minStep=" << tempMinStep << " ns\n";
134  }
135 
136  // Calculate the number of steps.
137  numSteps = (tempStop - tempStart) / tempStep;
138  if((tempStop - tempStart) % tempStep != 0)
139  numSteps++;
140 
141  if(debug){
142  std::cout << "steps=" << numSteps << std::endl;
143  for(int i = 0; i <= numSteps; i++)
144  std::cout << " " << i << ": " << startVal + i*stepSize << " ns\n";
145  }
146 
147  stopVal = startVal + numSteps * stepSize;
148  }
149 };
