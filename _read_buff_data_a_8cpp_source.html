67 int ReadBuffDataA(word_t *buf, unsigned long *bufLen,
68  vector<ChanEvent*> &eventList) {
69  unsigned long bufSkippedWords;
70  word_t evtPattern;
71  word_t bufNData, modNum, runTask, runStartTime[3], eventTime[2];
72  word_t chanTrigTime = 0, eventEnergy = 0, chanLength = 0;
73 
74  /* Initialize indicator and counter */
75  unsigned long totalSkippedWords = 0;
76  unsigned long numEvents = 0;
77 
78  static const double HIGH_MULT = pow(2., 32.);
79 
80  /* Determine the number of words in the buffer */
81  *bufLen = bufNData = buf[totalSkippedWords++];
82 
83  /* Read the module number */
84  modNum=buf[totalSkippedWords++];
85 
86  /* Read Run Task */
87  runTask=buf[totalSkippedWords++] & 0x0FFF;
88 
89  /* Read Run Start Time */
90  runStartTime[0]=buf[totalSkippedWords];
91  runStartTime[1]=buf[totalSkippedWords+1];
92  runStartTime[2]=buf[totalSkippedWords+2];
93  totalSkippedWords += 3;
94 
95  if( bufNData > BUFFER_HEAD_LENGTH ) { /* Check if buffer contains event */
96  /* Loop over the buffer */
97  bufSkippedWords = 0;
98  numEvents=0;
99  do {
100  /* Read Event Pattern */
101  evtPattern=buf[totalSkippedWords];
102  /* Read the Event Time */
103  eventTime[0]=buf[totalSkippedWords+1];
104  eventTime[1]=buf[totalSkippedWords+2];
105 
106  totalSkippedWords += EVENT_HEAD_LENGTH;
107  bufSkippedWords += EVENT_HEAD_LENGTH;
108 
109  if (bufSkippedWords > bufNData) {
110  cout << "Bad buffer " << totalSkippedWords << endl;
111  return readbuff::ERROR;
112  }
113 
114  if( evtPattern != 0 ) {
115  for(int ch = 0; ch < NUMBER_OF_CHANNELS; ch++) {
116  /* Check if Channel j has been hit */
117  if( pixie::TstBit(ch, evtPattern) ) {
118  /* Read Channel Length */
119  if( runTask == LIST_MODE_RUN0 || runTask == LIST_MODE_RUN1 ) {
120  chanLength=buf[totalSkippedWords];
121 
122  if( chanLength>10000 ) {
123  cout << "Bad ChanLen " << chanLength << endl;
124  return readbuff::ERROR;
125  }
126 
127  /* Read Channel trigger time */
128  chanTrigTime=buf[totalSkippedWords+1];
129  /* Read Channel event energy */
130  eventEnergy=buf[totalSkippedWords+2];
131  /* Skip the remaining channel header data */
132  totalSkippedWords += CHANNEL_HEAD_LENGTH;
133  bufSkippedWords += CHANNEL_HEAD_LENGTH;
134  } else if ( runTask == LIST_MODE_RUN3 ) {
135  /* Read Channel trigger time */
136  chanTrigTime=buf[totalSkippedWords];
137  /* Read Channel event energy */
138  eventEnergy=buf[totalSkippedWords+1];
139  totalSkippedWords += 2;
140  bufSkippedWords += 2;
141  } else {
142  cout << "Run task read out is invalid " << runTask << endl;
143  return readbuff::ERROR;
144  }
145  ChanEvent *currentEvt = new ChanEvent();
146 
147  currentEvt->chanNum = ch;
148  currentEvt->modNum = modNum;
149  if ( (eventEnergy & 0xFFFF0000) == 0) {
150  currentEvt->energy = eventEnergy;
151  } else {
152  // this is some peculiar pixie event, tag with a fixed energy
153  currentEvt->energy = 64000;
154  }
155  currentEvt->trigTime = chanTrigTime;
156  currentEvt->eventTimeHi = eventTime[0];
157  currentEvt->eventTimeLo = eventTime[1];
158  currentEvt->runTime0 = runStartTime[0];
159  currentEvt->runTime1 = runStartTime[1];
160  currentEvt->runTime2 = runStartTime[2];
161 
162  // new event time constructed from upper 32 bits of event time and
163  // using the trigger time for the lower 32 bits, this should
164  // be immune to slow filter lengths.
165  currentEvt->time = (double)(eventTime[0] * HIGH_MULT + chanTrigTime);
166 
167  /* Check if trace data follows the channel header */
168  if( chanLength > CHANNEL_HEAD_LENGTH && runTask == LIST_MODE_RUN0 ) {
169  halfword_t *hbuf = (halfword_t *)&buf[totalSkippedWords];
170  // Read the trace data (2-bytes per sample, i.e. 2 samples per word)
171  int numSamples = 2 * (chanLength - CHANNEL_HEAD_LENGTH);
172  for(int k = 0; k < numSamples; k ++) {
173  currentEvt->trace.push_back(hbuf[k]);
174  }
175 
176  totalSkippedWords += chanLength - CHANNEL_HEAD_LENGTH;
177  bufSkippedWords += chanLength - CHANNEL_HEAD_LENGTH;
178  }
179  eventList.push_back(currentEvt);
180  } // if channel hit
181  } // check channel hitpattern
182  } else { // if non-0 hitpattern
183  //patch 07/04/2006 for the events with pattern=0
184  ChanEvent *currentEvt = new ChanEvent;
185 
186  currentEvt->modNum = modNum;
187  currentEvt->chanNum = -1; // used to be set to 0, with id set to -1
188  currentEvt->energy = 65535;
189  currentEvt->trigTime = chanTrigTime;
190  currentEvt->eventTimeHi = eventTime[0];
191  currentEvt->eventTimeLo = eventTime[1];
192  currentEvt->runTime0 = runStartTime[0];
193  currentEvt->runTime1 = runStartTime[1];
194  currentEvt->runTime2 = runStartTime[2];
195  currentEvt->time = 0;
196 
197  eventList.push_back(currentEvt);
198  }
199  numEvents++;
200  } while( bufSkippedWords < (bufNData - BUFFER_HEAD_LENGTH) );
201  } else {// if buffer has data
202  cout << "ERROR BufNData " << bufNData << endl;
203  cout << "ERROR IN ReadBuffData" << endl;
204  cout << "LIST UNKNOWN" << endl;
205  return readbuff::ERROR;
206  }
207 
208  return numEvents;
209 }
