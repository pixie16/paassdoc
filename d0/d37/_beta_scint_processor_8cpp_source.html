145 bool BetaScintProcessor::Process(RawEvent &event) {
146  if (!EventProcessor::Process(event))
147  return false;
148 
149  static const vector<ChanEvent*> &scintBetaEvents =
150  event.GetSummary("beta_scint:beta", true)->GetList();
151 
152  double clockInSeconds = Globals::get()->clockInSeconds();
153 
155  bool tapeMove = !(TreeCorrelator::get()->place("Cycle")->status());
156 
158  double cycleTime = TreeCorrelator::get()->place("Cycle")->last().time;
159 
161  int multiplicityThres = 0;
162  int multiplicityGamma = 0;
163  for (vector<ChanEvent*>::const_iterator it = scintBetaEvents.begin();
164  it != scintBetaEvents.end(); it++) {
165  double energy = (*it)->GetCalEnergy();
166  int energyBin = int(energy / energyContraction_);
167  double time = (*it)->GetTime();
168  int location = (*it)->GetChanID().GetLocation();
169 
170  PlaceOR* betas =
171  dynamic_cast<PlaceOR*>(TreeCorrelator::get()->place("Beta"));
172  /* Beta events gated by "Beta" place are plotted here
173  * Energy-time spectra are gated
174  * */
175  for (deque<EventData>::iterator itb = betas->info_.begin();
176  itb != betas->info_.end(); ++itb) {
177  if (itb->energy == energy && itb->time == time &&
178  itb->location == location) {
179  ++multiplicityThres;
180  plot(D_ENERGY_BETA_THRES_GATED, energyBin);
181  //Break the deque loop since we found the matching event
182  break;
183  }
184  }
185 
186  //Skip the energy-time spectra for zero energy events
187  if (energyBin < 1)
188  continue;
189 
190  double decayTime = (time - cycleTime) * clockInSeconds;
191  int decayTimeBin = int(decayTime / timeSpectraTimeResolution);
192 
193  EventData bestGamma = BestGammaForBeta(time);
194  double gb_dtime = (time - bestGamma.time) * clockInSeconds;
195 
196  if (GoodGammaBeta(gb_dtime)) {
197  plot(D_ENERGY_BETA_GAMMA_GATED, energyBin);
198  plot(DD_ENERGY_BETA__GAMMA, energyBin, bestGamma.energy);
199  ++multiplicityGamma;
200  }
201 
202  if (tapeMove) {
203  plot(DD_ENERGY_BETA__TIME_TM_TOTAL,
204  energyBin, decayTimeBin);
205  if (GoodGammaBeta(gb_dtime))
206  plot(DD_ENERGY_BETA__TIME_TM_G,
207  energyBin, decayTimeBin);
208  else
209  plot(DD_ENERGY_BETA__TIME_TM_NOG,
210  energyBin, decayTimeBin);
211  } else {
212  plot(DD_ENERGY_BETA__TIME_TOTAL,
213  energyBin, decayTimeBin);
214  if (GoodGammaBeta(gb_dtime))
215  plot(DD_ENERGY_BETA__TIME_G,
216  energyBin, decayTimeBin);
217  else
218  plot(DD_ENERGY_BETA__TIME_NOG,
219  energyBin, decayTimeBin);
220  }
221  }
222 
223  plot(D_MULT_BETA_THRES_GATED, multiplicityThres);
224  plot(D_MULT_BETA_GAMMA_GATED, multiplicityGamma);
225 
226  EndProcess();
227  return(true);
228 }
