44 bool Unpacker::BuildRawEvent(){
45  if(!rawEvent.empty())
46  ClearRawEvent();
47 
48  if(numRawEvt == 0){// This is the first rawEvent. Do some special processing.
49  // Find the first XiaData event. The eventList is time sorted by module.
50  // The first component of each deque will be the earliest time from that module.
51  // The first event time will be the minimum of these first components.
52  if(!GetFirstTime(firstTime))
53  return false;
54  std::cout << "BuildRawEvent: First event time is " << firstTime << " clock ticks.\n";
55  eventStartTime = firstTime;
56  }
57  else{
58  // Move the event window forward to the next valid channel fire.
59  if(!GetFirstTime(eventStartTime))
60  return false;
61  }
62 
63  realStartTime = eventStartTime+eventWidth;
64  realStopTime = eventStartTime;
65 
66  unsigned int mod, chan;
67  string type, subtype, tag;
68  XiaData *current_event = NULL;
69 
70  // Loop over all time-sorted modules.
71  for(vector<deque<XiaData*> >::iterator iter = eventList.begin(); iter != eventList.end(); iter++){
72  if(iter->empty())
73  continue;
74 
75  // Loop over the list of channels that fired in this buffer
76  while(!iter->empty()){
77  current_event = iter->front();
78  mod = current_event->GetModuleNumber();
79  chan = current_event->GetChannelNumber();
80 
81  if(mod > MAX_PIXIE_MOD || chan > MAX_PIXIE_CHAN){ // Skip this channel
82  cout << "BuildRawEvent: Encountered non-physical Pixie ID (mod = " << mod << ", chan = " << chan << ")\n";
83  delete current_event;
84  iter->pop_front();
85  continue;
86  }
87 
88  double currtime = current_event->GetTime();
89 
90  // Check for backwards time-skip. This is un-handled currently and needs fixed CRT!!!
91  if(currtime < eventStartTime)
92  cout << "BuildRawEvent: Detected backwards time-skip from "
93  "start=" << eventStartTime << " to "
94  << current_event->GetTime() << "???\n";
95 
96  // If the time difference between the current and previous event is
97  // larger than the event width, finalize the current event, otherwise
98  // treat this as part of the current event
99  if((currtime - eventStartTime) > eventWidth){ // 62 pixie ticks represents ~0.5 us
100  break;
101  }
102 
103  // Check for the minimum time in this raw event.
104  if(currtime < realStartTime)
105  realStartTime = currtime;
106 
107  // Check for the maximum time in this raw event.
108  if(currtime > realStopTime)
109  realStopTime = currtime;
110 
111  // Update raw stats output with the new event before adding it to the raw event.
112  RawStats(current_event);
113 
114  // Push this channel event into the rawEvent.
115  rawEvent.push_back(current_event);
116 
117  // Remove this event from the event list but do not delete it yet.
118  // Deleting of the channel events will be handled by clearing the rawEvent.
119  iter->pop_front();
120  }
121  }
122 
123  numRawEvt++;
124 
125  return true;
126 }
