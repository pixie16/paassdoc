114  continue;
115  }
116 
117  // messy for right now
118  if (icEvents.size() == noDets) {

131  }
132 
133  }

136  plot(D_DTIME_DETX + loc, dtime / 10);
137 
138  if (dtime > minTime)
139  timeDiffs[loc].push_back( dtime );
140  if (timeDiffs[loc].size() >= timesToKeep) {
141  timeDiffs[loc].pop_front();
142 
143  // now calculate the rate as the inverse of the mean
144  // of the assumed Poissonic distribution
145  // since there is some dead time only take times greater than a
146  // specific safe value (thanks to memorylessness of distribution)
147

152  dit < timeDiffs[loc].end(); dit++) {
153  sum += *dit - minTime;
154  count++;
155  }
156  double mean = double(sum / count);
157  plot(D_RATE_DETX + loc,
158  (double)(1 / mean / Globals::get()->clockInSeconds()));
159  }
160 
161  }

163  }
