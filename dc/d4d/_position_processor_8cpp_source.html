60 bool PositionProcessor::Init(RawEvent& rawev)
61 {
62  if (!EventProcessor::Init(rawev))
63  return false;
64 
65  DetectorLibrary* modChan = DetectorLibrary::get();
66 
67  int numLocationsTop = modChan->GetNextLocation("ssd", "top");
68  int numLocationsBottom = modChan->GetNextLocation("ssd", "bottom");
69  if (numLocationsTop != numLocationsBottom) {
70  cerr << "Number of top positions (" << numLocationsTop
71  << ") does not match number of bottom positions ("
72  << numLocationsBottom << ") in map!" << endl;
73  cerr << " Disabling QDC processor." << endl;
74 
75  return (initDone = false);
76  }
77  numLocations = numLocationsTop;
78  if (numLocations > maxNumLocations) {
79  cerr << "Number of positions (" << numLocations
80  << ") is larger then maximum number of supported positions ("
81  << maxNumLocations << ") in PositionProcessor" << endl;
82  cerr << " Disabling QDC processor." << endl;
83  return (initDone = false);
84  }
85  minNormQdc.resize(numLocations);
86  maxNormQdc.resize(numLocations);
87 
88  //This functionality is deprecated and should be implemented in the XML file
89  //string configFile = Globals::get()->configPath("qdc.txt");
90  string configFile = "";
91 
92  ifstream in(configFile.c_str());
93  if (!in) {
94  cerr << "Failed to open the QDC parameter file, QDC processor disabled." << endl;
95  return (initDone = false);
96  }
97 
98  // Ignore any lines at the beginning that don't have a digit
99  // This allows for some comments at the beginning of the file
100  int linesIgnored = 0;
101  while ( !isdigit(in.peek()) ) {
102  in.ignore(1000, '\n');
103  linesIgnored++;
104  }
105  if (linesIgnored != 0) {
106  cout << "Ignored " << linesIgnored << " comment lines in "
107  << configFile << endl;
108  }
109 
110  for (int i=0; i < numQdcs; i++)
111  in >> qdcLen[i];
112  partial_sum(qdcLen, qdcLen + numQdcs, qdcPos);
113  totLen = qdcPos[numQdcs - 1] - qdcLen[0];
114  // totLen = accumulate(qdcLen + 1, qdcLen + 8, 0);
115 
116  in >> whichQdc >> posScale;
117 
118  int numLocationsRead = 0;
119  while (true) {
120  int location;
121  in >> location;
122  if (in.eof()) {
123  // place this here so a trailing newline is okay in the scan file
124  break;
125  }
126  in >> minNormQdc[location] >> maxNormQdc[location];
127  numLocationsRead++;
128  }
129 
130  if (numLocationsRead != numLocations) {
131  cerr << "Only read QDC position calibration information from "
132  << numLocationsRead << " locations where a total of "
133  << numLocations << " was expected!" << endl;
134  cerr << " Disabling position processor." << endl;
135  return (initDone = false);
136  }
137 
138  cout << "QDC processor initialized with " << numLocations
139  << " locations operating on " << numQdcs << " QDCs" << endl;
140  cout << "QDC #" << whichQdc << " being used for position determination."
141  << endl;
142 
143  return true;

234  return false;

237  event.GetSummary("ssd:sum", true)->GetList();
238  static const vector<ChanEvent*> &digisumEvents =
239  event.GetSummary("ssd:digisum", true)->GetList();
240  static const vector<ChanEvent*> &topEvents =
241  event.GetSummary("ssd:top", true)->GetList();
242  static const vector<ChanEvent*> &bottomEvents =
243  event.GetSummary("ssd:bottom", true)->GetList();
244 
245  vector<ChanEvent *> allEvents;
246  // just add in the digisum events for now
247  allEvents.insert(allEvents.begin(), digisumEvents.begin(), digisumEvents.end());
248  allEvents.insert(allEvents.begin(), sumEvents.begin(), sumEvents.end());
249 
250  for (vector<ChanEvent*>::const_iterator it = allEvents.begin();
251  it != allEvents.end(); ++it) {
252  ChanEvent *sumchan = *it;
253 
254  int location = sumchan->GetChanID().GetLocation();
255 
256  // Don't waste our time with noise events
257  if ( (*it)->GetEnergy() < 10. || (*it)->GetEnergy() > 16374 ) {
258  using namespace dammIds::position;
259 
260  plot(D_INFO_LOCX + location, INFO_NOISE);
261  plot(D_INFO_LOCX + LOC_SUM , INFO_NOISE);
262  continue;
263  }
264 
265  const ChanEvent *top = FindMatchingEdge(sumchan, topEvents.begin(), topEvents.end());
266  const ChanEvent *bottom = FindMatchingEdge(sumchan, bottomEvents.begin(), bottomEvents.end());
267 
268 
269  if (top == NULL || bottom == NULL) {
270  using namespace dammIds::position;
271 
272  if (top == NULL) {
273  // [6] -> Missing top
274  plot(D_INFO_LOCX + location, INFO_MISSING_TOP);
275  plot(D_INFO_LOCX + LOC_SUM, INFO_MISSING_TOP);
276  }
277 
278  if (bottom == NULL) {
279  // [5] -> Missing bottom
280  plot(D_INFO_LOCX + location, INFO_MISSING_BOTTOM);
281  plot(D_INFO_LOCX + LOC_SUM, INFO_MISSING_BOTTOM);
282  }
283  continue;
284  }
285 
286  /* Make sure we get the same match going backwards to insure there is only one in the vector */
287  if ( FindMatchingEdge(sumchan, topEvents.rbegin(), topEvents.rend()) != top) {
288  using namespace dammIds::position;
289  // [4] -> Multiple top
290  plot(D_INFO_LOCX + location, INFO_MULTIPLE_TOP);
291  plot(D_INFO_LOCX + LOC_SUM, INFO_MULTIPLE_TOP);
292  continue;
293  }
294  if ( FindMatchingEdge(sumchan, bottomEvents.rbegin(), bottomEvents.rend()) != bottom) {
295  using namespace dammIds::position;
296  // [3] -> Multiple bottom
297  plot(D_INFO_LOCX + location, INFO_MULTIPLE_BOTTOM);
298  plot(D_INFO_LOCX + LOC_SUM, INFO_MULTIPLE_BOTTOM);
299  continue;
300  }
301 
302  using namespace dammIds::position;
303 
304  float topQdc[numQdcs];
305  float bottomQdc[numQdcs];
306  float topQdcTot = 0;
307  float bottomQdcTot = 0;
308  float position = NAN;
309 
310  topQdc[0] = top->GetQdcValue(0);
311  bottomQdc[0] = bottom->GetQdcValue(0);
312  if (bottomQdc[0] == pixie::U_DELIMITER || topQdc[0] == pixie::U_DELIMITER) {
313  // This happens naturally for traces which have double triggers
314  // Onboard DSP does not write QDCs in this case
315 #ifdef VERBOSE
316  cout << "SSD strip edges are missing QDC information for location " << location << endl;
317 #endif
318  if (topQdc[0] == pixie::U_DELIMITER) {
319  // [2] -> Missing top QDC
320  plot(D_INFO_LOCX + location, INFO_MISSING_TOP_QDC);
321  plot(D_INFO_LOCX + LOC_SUM, INFO_MISSING_TOP_QDC);
322  // Recreate qdc from trace
323  if ( !top->GetTrace().empty() ) {
324  topQdc[0] = accumulate(top->GetTrace().begin(), top->GetTrace().begin() + qdcLen[0], 0);
325  } else {
326  topQdc[0] = 0;
327  }
328  }
329  if (bottomQdc[0] == pixie::U_DELIMITER) {
330  // [1] -> Missing bottom QDC
331  plot(D_INFO_LOCX + location, INFO_MISSING_BOTTOM_QDC);
332  plot(D_INFO_LOCX + LOC_SUM, INFO_MISSING_BOTTOM_QDC);
333  // Recreate qdc from trace
334  if ( !bottom->GetTrace().empty() ) {
335  bottomQdc[0] = accumulate(bottom->GetTrace().begin(), bottom->GetTrace().begin() + qdcLen[0], 0);
336  } else {
337  bottomQdc[0] = 0;
338  }
339  }
340  if ( topQdc[0] == 0 || bottomQdc[0] == 0 ) {
341  continue;
342  }
343  }
344  // [0] -> good stuff
345  plot(D_INFO_LOCX + location, INFO_OKAY);
346  plot(D_INFO_LOCX + LOC_SUM, INFO_OKAY);
347 
348 
349  for (int i = 1; i < numQdcs; ++i) {
350  if (top->GetQdcValue(i) == pixie::U_DELIMITER) {
351  // Recreate qdc from trace
352  topQdc[i] = accumulate(top->GetTrace().begin() + qdcPos[i-1],
353  top->GetTrace().begin() + qdcPos[i], 0);
354  } else {
355  topQdc[i] = top->GetQdcValue(i);
356  }
357 
358  topQdc[i] -= topQdc[0] * qdcLen[i] / qdcLen[0];
359  topQdcTot += topQdc[i];
360  topQdc[i] /= qdcLen[i];
361 
362  if (bottom->GetQdcValue(i) == pixie::U_DELIMITER) {
363  // Recreate qdc from trace
364  bottomQdc[i] = accumulate(bottom->GetTrace().begin() + qdcPos[i-1],
365  bottom->GetTrace().begin() + qdcPos[i], 0);
366  } else {
367  bottomQdc[i] = bottom->GetQdcValue(i);
368  }
369 
370  bottomQdc[i] -= bottomQdc[0] * qdcLen[i] / qdcLen[0];
371  bottomQdcTot += bottomQdc[i];
372  bottomQdc[i] /= qdcLen[i];
373 
374  plot(DD_QDCN__QDCN_LOCX + QDC_JUMP * i + location, topQdc[i] + 10, bottomQdc[i] + 10);
375  plot(DD_QDCN__QDCN_LOCX + QDC_JUMP * i + LOC_SUM, topQdc[i], bottomQdc[i]);
376 
377  float frac = topQdc[i] / (topQdc[i] + bottomQdc[i]) * 1000.; // per mil
378 
379  plot(D_QDCNORMN_LOCX + QDC_JUMP * i + location, frac);
380  plot(D_QDCNORMN_LOCX + QDC_JUMP * i + LOC_SUM, frac);
381  if (i == whichQdc) {
382  position = posScale * (frac - minNormQdc[location]) /
383  (maxNormQdc[location] - minNormQdc[location]);
384  sumchan->GetTrace().InsertValue("position", position);
385  plot(DD_POSITION__ENERGY_LOCX + location, position, sumchan->GetCalEnergy());
386  plot(DD_POSITION__ENERGY_LOCX + LOC_SUM, position, sumchan->GetCalEnergy());
387  }
388  if (i == 6 && !sumchan->IsSaturated()) {
389  // compare the long qdc to the energy
390  int qdcSum = topQdc[i] + bottomQdc[i];
391 
392  // MAGIC NUMBERS HERE, move to qdc.txt
393  if (qdcSum < 1000 && sumchan->GetCalEnergy() > 15000) {
394  sumchan->GetTrace().InsertValue("badqdc", 1);
395  } else if ( !isnan(position) ) {
396  plot(DD_POSITION, location, position);
397  }
398  }
399  } // end loop over qdcs
400  // KM QDC - QDC correlations
401  double ratio[4] = {0};
402  for (int i = 1; i < 5; ++i) {
403  ratio[i - 1] = topQdc[i] / (bottomQdc[i] + topQdc[i]) * 1000.0;
404  }
405 
406  plot(DD_QDCR2__QDCR1_LOCX + location, ratio[1], ratio[0]);
407  plot(DD_QDCR2__QDCR3_LOCX + location, ratio[1], ratio[2]);
408  plot(DD_QDCR2__QDCR4_LOCX + location, ratio[1], ratio[3]);
409 
410  plot(DD_QDC1R__POS_LOCX + location, ratio[0], position * 10.0 + 200.0);
411  plot(DD_QDC2R__POS_LOCX + location, ratio[1], position * 10.0 + 200.0);
412  plot(DD_QDC3R__POS_LOCX + location, ratio[2], position * 10.0 + 200.0);
413  plot(DD_QDC4R__POS_LOCX + location, ratio[3], position * 10.0 + 200.0);
414 
415  topQdcTot /= totLen;
416  bottomQdcTot /= totLen;
417 
418  plot(DD_QDCTOT__QDCTOT_LOCX + location, topQdcTot, bottomQdcTot);
419  plot(DD_QDCTOT__QDCTOT_LOCX + LOC_SUM, topQdcTot, bottomQdcTot);
420  } // end iteration over sum events
421 
422  EndProcess();
423 
424  return true;
