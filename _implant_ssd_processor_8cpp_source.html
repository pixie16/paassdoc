457 EventInfo::EEventTypes ImplantSsdProcessor::SetType(EventInfo &info) const
458 {
459  if (info.hasVeto) {
460  return (info.type = EventInfo::PROTON_EVENT);
461  }
462 
463  bool enoughMcp = (info.mcpMult >= info.impMult);
464  // high energy events
465  if (info.energy > cutoffEnergy) {
466  if ( info.beamOn) {
467  if (info.hasTof && enoughMcp) {
468  if (info.tof > implantTof) {
469  return (info.type = EventInfo::IMPLANT_EVENT);
470  }
471  return (info.type = EventInfo::PROJECTILE_EVENT);
472  } else {
473  // NO TAC OR MCP
474  if (info.energy > fissionThresh && !info.hasTof && !enoughMcp)
475 
476  return (info.type = EventInfo::FISSION_EVENT);
477  // Implant with lost TAC or MCP possibly?
478  return (info.type = EventInfo::UNKNOWN_EVENT);
479  }
480  }
481  // NO BEAM
482  if (info.hasTof || enoughMcp) {
483  return (info.type = EventInfo::UNKNOWN_EVENT);
484  }
485  if (info.energy > fissionThresh) {
486  return (info.type = EventInfo::FISSION_EVENT);
487  }
488  return (info.type = EventInfo::UNKNOWN_EVENT);
489  }
490 
491  // low energy events
492  if (info.pileUp && !enoughMcp) {
493  return (info.type = EventInfo::DECAY_EVENT);
494  }
495  if (!info.pileUp && !enoughMcp && !info.hasTof) {
496  return (info.type = EventInfo::DECAY_EVENT);
497  }
498  if (info.beamOn && (info.hasTof || enoughMcp) )
499  return (info.type = EventInfo::PROJECTILE_EVENT);
500 
501  return (info.type = EventInfo::UNKNOWN_EVENT);
502 }
