73 int ReadBuffDataD(word_t *buf, unsigned long *bufLen,
74  vector<ChanEvent*> &eventList) {
75  // multiplier for high bits of 48-bit time
76  static const double HIGH_MULT = pow(2., 32.);
77  word_t modNum;
78  unsigned long numEvents = 0;
79  word_t *bufStart = buf;
80  /* Determine the number of words in the buffer */
81  *bufLen = *buf++;
82  /* Read the module number */
83  modNum = *buf++;
84  ChanEvent *lastVirtualChannel = NULL;
85  if(*bufLen > 0) { // check if the buffer has data
86  if(*bufLen == 2) {
87  // this is an empty channel
88  return 0;
89  }
90  do {
91  ChanEvent *currentEvt = new ChanEvent;
92  // decoding event data... see pixie16app.c
93  // buf points to the start of channel data
94  word_t chanNum = (buf[0] & 0x0000000F);
95  word_t slotNum = (buf[0] & 0x000000F0) >> 4;
96  word_t crateNum = (buf[0] & 0x00000F00) >> 8;
97  word_t headerLength = (buf[0] & 0x0001F000) >> 12;
98  word_t eventLength = (buf[0] & 0x1FFE0000) >> 17;
99  currentEvt->virtualChannel = ((buf[0] & 0x20000000) != 0);
100  currentEvt->saturatedBit = ((buf[0] & 0x40000000) != 0);
101  currentEvt->pileupBit = ((buf[0] & 0x80000000) != 0);
102  // Rev. D header lengths not clearly defined in pixie16app_defs
104  // make some sanity checks
105  if(headerLength == stats.headerLength) {
106  // this is a manual statistics block inserted by the poll program
107  stats.DoStatisticsBlock(&buf[1], modNum);
108  buf += eventLength;
109  numEvents = readbuff::STATS;
110  continue;
111  }
112  if(headerLength != 4 && headerLength != 8 &&
113  headerLength != 12 && headerLength != 16) {
114  cout << " Unexpected header length: " << headerLength << endl;
115  cout << " Buffer " << modNum << " of length " << *bufLen << endl;
116  cout << " CHAN:SLOT:CRATE "
117  << chanNum << ":" << slotNum << ":" << crateNum << endl;
118  // advance to next event and continue
119  // buf += EventLength;
120  // continue;
121  // skip the rest of this buffer
122  return readbuff::ERROR;
123  //return numEvents;
124  }
125  word_t lowTime = buf[1];
126  word_t highTime = buf[2] & 0x0000FFFF;
127  word_t cfdTime = (buf[2] & 0xFFFF0000) >> 16;
128  word_t energy = buf[3] & 0x0000FFFF;
129  word_t traceLength = (buf[3] & 0xFFFF0000) >> 16;
130  if(headerLength == 8 || headerLength == 16) {
131  // skip the onboard partial sums for now
132  // trailing, leading, gap, baseline
133  }
134  if(headerLength >= 12) {
135  int offset = headerLength - 8;
136  for(int i=0; i < currentEvt->numQdcs; i++) {
137  currentEvt->qdcValue[i] = buf[offset + i];
138  }
139  }
140  // one last sanity check
141  if(traceLength / 2 + headerLength != eventLength) {
142  cout << " Bad event length (" << eventLength
143  << ") does not correspond with length of header (" << headerLength
144  << ") and length of trace (" << traceLength << ")" << endl;
145  buf += eventLength;
146  continue;
147  }
148  // handle multiple crates
149  modNum += 100 * crateNum;
150  currentEvt->chanNum = chanNum;
151  currentEvt->modNum = modNum;
152  if(currentEvt->virtualChannel) {
153  DetectorLibrary* modChan = DetectorLibrary::get();
154  currentEvt->modNum += modChan->GetPhysicalModules();
155  if(modChan->at(modNum, chanNum).HasTag("construct_trace")) {
156  lastVirtualChannel = currentEvt;
157  }
158  }
159  currentEvt->energy = energy;
160  //KM 2012-10-24 reinstating removal of saturated
161  if(currentEvt->saturatedBit)
162  currentEvt->energy = 16383;
163  currentEvt->trigTime = lowTime;
164  currentEvt->cfdTime = cfdTime;
165  currentEvt->eventTimeHi = highTime;
166  currentEvt->eventTimeLo = lowTime;
167  currentEvt->time = highTime * HIGH_MULT + lowTime;
168  buf += headerLength;
169  /* Check if trace data follows the channel header */
170  if(traceLength > 0) {
171  // sbuf points to the beginning of trace data
172  halfword_t *sbuf = (halfword_t *)buf;
173  currentEvt->trace.reserve(traceLength);
174  if(currentEvt->saturatedBit)
175  currentEvt->trace.SetValue("saturation", 1);
176  if(lastVirtualChannel != NULL && lastVirtualChannel->trace.empty()) {
177  lastVirtualChannel->trace.assign(traceLength, 0);
178  }
179  // Read the trace data (2-bytes per sample, i.e. 2 samples per word)
180  for(unsigned int k = 0; k < traceLength; k ++) {
181  currentEvt->trace.push_back(sbuf[k]);
182  if(lastVirtualChannel != NULL) {
183  lastVirtualChannel->trace[k] += sbuf[k];
184  }
185  }
186  buf += traceLength / 2;
187  }
188  eventList.push_back(currentEvt);
189  numEvents++;
190  } while(buf < bufStart + *bufLen);
191  } else { // if buffer has data
192  cout << "ERROR BufNData " << *bufLen << endl;
193  cout << "ERROR IN ReadBuffData" << endl;
194  cout << "LIST UNKNOWN" << endl;
195  return readbuff::ERROR;
196  }
197  return numEvents;
198 }
