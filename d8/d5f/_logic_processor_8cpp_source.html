135  double time = chan->GetTime();
136 
137  static double t0 = time;
138 
139  // for 2d plot of events 100ms / bin
140  const double eventsResolution = 100e-3 / clockInSeconds;
141  const unsigned MTC_START = 0;
142  const unsigned MTC_STOP = 1;
143  const unsigned BEAM_START = 2;
144  const unsigned BEAM_STOP = 3;
145  // for 2d plot of events 1s / bin
146  // Bins in plot
147  const unsigned BEAM_TOGGLE = 0;
148  const unsigned BEAM_ANALOG = 1;
149  const unsigned BEAM_NONE = 2;
150  double time_x = int((time - t0) / eventsResolution);
151 
152  if(subtype == "start") {
153  if (!std::isnan(lastStartTime.at(loc))) {
154  double timediff = time - lastStartTime.at(loc);
155  plot(DD_TDIFF_START, timediff / logicPlotResolution, loc);
156  plot(DD_TDIFF_SUM, timediff / logicPlotResolution, loc);
157  }
158 
159  lastStartTime.at(loc) = time;
160  logicStatus.at(loc) = true;
161 
162  startCount.at(loc)++;
163  plot(D_COUNTER_START, loc);
164  } else if (subtype == "stop") {
165  if (!std::isnan(lastStopTime.at(loc))) {
166  double timediff = time - lastStopTime.at(loc);
167  plot(DD_TDIFF_STOP, timediff / logicPlotResolution, loc);
168  plot(DD_TDIFF_SUM, timediff / logicPlotResolution, loc);
169  if (!std::isnan(lastStartTime.at(loc))) {
170  double moveTime = time - lastStartTime.at(loc);
171  plot(DD_TDIFF_LENGTH, moveTime / logicPlotResolution, loc);
172  }
173  }
174 
175  lastStopTime.at(loc) = time;
176  logicStatus.at(loc) = false;
177 
178  stopCount.at(loc)++;
179  plot(D_COUNTER_STOP, loc);
180  } else if(place == "logic_mtc_start_0") {
181  double dt_start = time -
182  TreeCorrelator::get()->place(place)->secondlast().time;
183  TreeCorrelator::get()->place("TapeMove")->activate(time);
184  TreeCorrelator::get()->place("Cycle")->deactivate(time);
185 
186  plot(D_TDIFF_MOVE_START, dt_start / mtcPlotResolution);
187  plot(D_COUNTER, MOVE_START_BIN);
188  plot(DD_TIME_DET_MTCEVENTS, time_x, MTC_START);
189  } else if (place == "logic_mtc_stop_0") {
190  double dt_stop = time -
191  TreeCorrelator::get()->place(place)->secondlast().time;
192  double dt_move = time -
193  TreeCorrelator::get()->place("logic_mtc_start_0")->last().time;
194  TreeCorrelator::get()->place("TapeMove")->deactivate(time);
195 
196  plot(D_TDIFF_MOVE_STOP, dt_stop / mtcPlotResolution);
197  plot(D_MOVETIME, dt_move / mtcPlotResolution);
198  plot(D_COUNTER, MOVE_STOP_BIN);
199  plot(DD_TIME_DET_MTCEVENTS, time_x, MTC_STOP);
200  } else if (place == "logic_beam_start_0") {
201  double dt_start = time -
202  TreeCorrelator::get()->place(place)->secondlast().time;
203  //Remove double starts
204  if (doubleStart_) {
205  double dt_stop =
206  abs(time -
207  TreeCorrelator::get()->place("logic_beam_stop_0")->last().time);
208  if (abs(dt_start * clockInSeconds) < doubleTimeLimit_ ||
209  abs(dt_stop * clockInSeconds) < doubleTimeLimit_)
210  continue;
211  }
212  TreeCorrelator::get()->place("Beam")->activate(time);
213  TreeCorrelator::get()->place("Cycle")->activate(time);
214 
215  plot(D_TDIFF_BEAM_START, dt_start / mtcPlotResolution);
216  plot(D_COUNTER, BEAM_START_BIN);
217  plot(DD_TIME_DET_MTCEVENTS, time_x, BEAM_START);
218  } else if (place == "logic_beam_stop_0") {
219  double dt_stop = time -
220  TreeCorrelator::get()->place(place)->secondlast().time;
221  double dt_beam = time -
222  TreeCorrelator::get()->place("logic_beam_start_0")->last().time;
223  //Remove double stops
224  if (doubleStop_) {
225  if (abs(dt_stop * clockInSeconds) < doubleTimeLimit_ ||
226  abs(dt_beam * clockInSeconds) < doubleTimeLimit_)
227  continue;
228  }
229  TreeCorrelator::get()->place("Beam")->deactivate(time);
230 
231  plot(D_TDIFF_BEAM_STOP, dt_stop / mtcPlotResolution);
232  plot(D_BEAMTIME, dt_beam / mtcPlotResolution);
233  plot(D_COUNTER, BEAM_STOP_BIN);
234  plot(DD_TIME_DET_MTCEVENTS, time_x, BEAM_STOP);
235  } else if (place == "logic_t1_0") {
236  //TreeCorrelator::get()->place("Protons")->activate(time);
237  double dt_t1 = time -
238  TreeCorrelator::get()->place("logic_t1_0")->last().time;
239  plot(D_TDIFF_T1, dt_t1 / mtcPlotResolution);
240  } else if (place == "logic_supercycle_0") {
241  TreeCorrelator::get()->place("Supercycle")->activate(time);
242  double dt_supercycle = time -
243  TreeCorrelator::get()->place("logic_supercycle_0")->last().time;
244  plot(D_TDIFF_SUPERCYCLE, dt_supercycle / mtcPlotResolution);
245  }else if (place == "logic_beam_0") {
246  double last_time =
247  TreeCorrelator::get()->place(place)->secondlast().time;
248  double dt_beam_stop = abs(time - last_time);
249 
250  Messenger m;
251  stringstream ss;
252  // Check for double recorded same event (1 us limit)
253  if (dt_beam_stop < doubleTimeLimit_) {
254  ss << "Ignore fast beam stop" << endl;
255  m.warning(ss.str());
256  continue;
257  }
258 
259  // If beam was stopped, activate place and plot stop length
260  if (!TreeCorrelator::get()->place("Beam")->status()) {
261  double clockInSeconds = Globals::get()->clockInSeconds();
262  double resolution = 1.0 / clockInSeconds;
263 
264  plot(D_TIME_STOP_LENGTH, dt_beam_stop / resolution);
265 
266  TreeCorrelator::get()->place("Beam")->activate(time);
267  ss << "Beam started after: " << dt_beam_stop / resolution
268  << " s ";
269  m.run_message(ss.str());
270  }
271  else {
272  TreeCorrelator::get()->place("Beam")->deactivate(time);
273  ss << "Beam stopped";
274  m.run_message(ss.str());
275  }
276  plot(D_COUNTER_BEAM, BEAM_TOGGLE);
277 
278  }
279  else if (place == "logic_analog_0") {
280  plot(D_COUNTER_BEAM, BEAM_ANALOG);
281  }
282  else if (place == "logic_none_0") {
283  plot(D_COUNTER_BEAM, BEAM_NONE);
284  }
285  }//events loop
286  return(true);
287 }
