342 
343  string place = (*it)->GetChanID().GetPlaceName();
344  if (place == "__9999")
345  continue;
346 
347  if ( (*it)->IsSaturated() || (*it)->IsPileup() )
348  continue;
349 
350  double time = (*it)->GetTime();

352  int location = (*it)->GetChanID().GetLocation();
353 
354  EventData data(time, energy, location);

356  }
357 
360  for (vector<EventProcessor*>::iterator iProc = vecProcess.begin();
361  iProc != vecProcess.end(); iProc++)
362  if ( (*iProc)->HasEvent() )
363  (*iProc)->PreProcess(rawev);
366  for (vector<EventProcessor *>::iterator iProc = vecProcess.begin();
367  iProc != vecProcess.end(); iProc++)
368  if ( (*iProc)->HasEvent() )
369  (*iProc)->Process(rawev);
370  // Clear all places in correlator (if of resetable type)
371  for (map<string, Place*>::iterator it =
372  TreeCorrelator::get()->places_.begin();
373  it != TreeCorrelator::get()->places_.end(); ++it)
374  if ((*it).second->resetable())
375  (*it).second->reset();
376  } catch (GeneralException &e) {
379  cout << endl

381  << endl;

460 int DetectorDriver::ThreshAndCal(ChanEvent *chan, RawEvent& rawev) {
461  Identifier chanId = chan->GetChanID();
462  int id = chan->GetID();
463  string type = chanId.GetType();
464  string subtype = chanId.GetSubtype();
465  map<string, int> tags = chanId.GetTagMap();
466  bool hasStartTag = chanId.HasTag("start");
467  Trace &trace = chan->GetTrace();
468 
469  RandomPool* randoms = RandomPool::get();
470 
471  double energy = 0.0;
472 
473  if (type == "ignore" || type == "")
474  return(0);
475 
476  if ( !trace.empty() ) {
477  plot(D_HAS_TRACE, id);
478 
479  for (vector<TraceAnalyzer *>::iterator it = vecAnalyzer.begin();
480  it != vecAnalyzer.end(); it++) {
481  (*it)->Analyze(trace, type, subtype, tags);
482  }
483 
484  if (trace.HasValue("filterEnergy") ) {
485  if (trace.GetValue("filterEnergy") > 0) {
486  energy = trace.GetValue("filterEnergy");
487  plot(D_FILTER_ENERGY + id, energy);
488  trace.SetValue("filterEnergyCal",
489  cali.GetCalEnergy(chanId, trace.GetValue("filterEnergy")));
490  } else {
491  energy = 0.0;
492  }
493 
496  int pulses = trace.GetValue("numPulses");
497  for (int i = 1; i < pulses; ++i) {
498  stringstream energyName;
499  energyName << "filterEnergy" << i + 1;
500  stringstream energyCalName;
501  energyCalName << "filterEnergy" << i + 1 << "Cal";
502  trace.SetValue(energyCalName.str(),
503  cali.GetCalEnergy(chanId,
504  trace.GetValue(energyName.str())));
505  }
506  }
507 
508  if (trace.HasValue("calcEnergy") ) {
509  energy = trace.GetValue("calcEnergy");
510  chan->SetEnergy(energy);
511  } else if (!trace.HasValue("filterEnergy")) {
512  energy = chan->GetEnergy() + randoms->Get();
513  }
514 
515  if (trace.HasValue("phase") ) {
516  //Saves the time in nanoseconds
517  chan->SetHighResTime((trace.GetValue("phase") *
518  Globals::get()->adcClockInSeconds() +
519  (double)chan->GetTrigTime() *
520  Globals::get()->filterClockInSeconds()) * 1e9);
521  }
522  } else {
526  energy = chan->GetEnergy() + randoms->Get();
527  chan->SetHighResTime(0.0);
528  }
529 
531  double time, walk_correction;
532  if(chan->GetHighResTime() == 0.0) {
533  time = chan->GetTime(); //time is in clock ticks
534  walk_correction = walk.GetCorrection(chanId, energy);
535  } else {
536  time = chan->GetHighResTime(); //time here is in ns
537  walk_correction = walk.GetCorrection(chanId, trace.GetValue("tqdc"));
538  }
539 
540  chan->SetCalEnergy(cali.GetCalEnergy(chanId, energy));
541  chan->SetCorrectedTime(time - walk_correction);
542 
543  rawev.GetSummary(type)->AddEvent(chan);
544  DetectorSummary *summary;
545 
546  summary = rawev.GetSummary(type + ':' + subtype, false);
547  if (summary != NULL)
548  summary->AddEvent(chan);
549 
550  if(hasStartTag && type != "logic") {
551  summary =
552  rawev.GetSummary(type + ':' + subtype + ':' + "start", false);
553  if (summary != NULL)
554  summary->AddEvent(chan);
555  }
556  return(1);
557 }

578 void DetectorDriver::ReadCalXml() {

592 
598  bool verbose = map.attribute("verbose_calibration").as_bool();
599  for (pugi::xml_node module = map.child("Module"); module;
600  module = module.next_sibling("Module")) {
601  int module_number = module.attribute("number").as_int(-1);
602  for (pugi::xml_node channel = module.child("Channel"); channel;
603  channel = channel.next_sibling("Channel")) {
604  int ch_number = channel.attribute("number").as_int(-1);
605  Identifier chanID = DetectorLibrary::get()->at(module_number,
606  ch_number);
607  bool calibrated = false;
608  for (pugi::xml_node cal = channel.child("Calibration");
609  cal; cal = cal.next_sibling("Calibration")) {
610  string model = cal.attribute("model").as_string("None");
611  double min = cal.attribute("min").as_double(0);
612  double max =

614 
615  stringstream pars(cal.text().as_string());
616  vector<double> parameters;
617  while (true) {
618  double p;
619  pars >> p;
620  if (pars)
621  parameters.push_back(p);
622  else
623  break;
624  }
625  if (verbose) {
626  stringstream ss;
627  ss << "Module " << module_number << ", channel "
628  << ch_number << ": ";
629  ss << " model-" << model;
630  for (vector<double>::iterator it = parameters.begin();
631  it != parameters.end(); ++it)
632  ss << " " << (*it);
633  m.detail(ss.str(), 1);
634  }
635  cali.AddChannel(chanID, model, min, max, parameters);
636  calibrated = true;
637  }
638  if (!calibrated && verbose) {
639  stringstream ss;
640  ss << "Module " << module_number << ", channel "

650 void DetectorDriver::ReadWalkXml() {
651  pugi::xml_document doc;
652 
653  pugi::xml_parse_result result = doc.load_file(cfg_.c_str());
654  if (!result) {
655  stringstream ss;
656  ss << "DetectorDriver: error parsing file " << cfg_;
657  ss << " : " << result.description();
658  throw GeneralException(ss.str());
659  }
660 
661  Messenger m;
662  m.start("Loading Walk Corrections");
663 
664  pugi::xml_node map = doc.child("Configuration").child("Map");
666  bool verbose = map.attribute("verbose_walk").as_bool();
667  for (pugi::xml_node module = map.child("Module"); module;
668  module = module.next_sibling("Module")) {
669  int module_number = module.attribute("number").as_int(-1);
670  for (pugi::xml_node channel = module.child("Channel"); channel;
671  channel = channel.next_sibling("Channel")) {
672  int ch_number = channel.attribute("number").as_int(-1);
673  Identifier chanID = DetectorLibrary::get()->at(module_number,
674  ch_number);
675  bool corrected = false;
676  for (pugi::xml_node walkcorr = channel.child("WalkCorrection");
677  walkcorr; walkcorr = walkcorr.next_sibling("WalkCorrection")) {
678  string model = walkcorr.attribute("model").as_string("None");
679  double min = walkcorr.attribute("min").as_double(0);
680  double max =
681  walkcorr.attribute("max").as_double(
682  numeric_limits<double>::max());
683 
684  stringstream pars(walkcorr.text().as_string());
685  vector<double> parameters;
686  while (true) {
687  double p;
688  pars >> p;
689  if (pars)
690  parameters.push_back(p);
691  else
692  break;
693  }
694  if (verbose) {
695  stringstream ss;
696  ss << "Module " << module_number
697  << ", channel " << ch_number << ": ";
698  ss << " model: " << model;
699  for (vector<double>::iterator it = parameters.begin();
700  it != parameters.end(); ++it)
701  ss << " " << (*it);
702  m.detail(ss.str(), 1);
703  }
704  walk.AddChannel(chanID, model, min, max, parameters);
705  corrected = true;
706  }
707  if (!corrected && verbose) {
708  stringstream ss;
709  ss << "Module " << module_number << ", channel "
710  << ch_number << ": ";
711  ss << " not corrected for walk";
712  m.detail(ss.str(), 1);
713  }
714  }
715  }
716  m.done();
717 }
